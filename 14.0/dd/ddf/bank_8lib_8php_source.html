295 function checkBanForAccount($account)
296 {
297  $country_code = $account->getCountryCode();
298 
299  // For compatibility between
300  // account of type CompanyBankAccount class (we use number, cle_rib)
301  // account of type Account class (we use num_compte, cle)
302  if (empty($account->number)) {
303  $account->number = $account->num_compte;
304  }
305  if (empty($account->cle)) {
306  $account->cle = $account->cle_rib;
307  }
308 
309  dol_syslog("bank.lib::checkBanForAccount account->code_banque=".$account->code_banque." account->code_guichet=".$account->code_guichet." account->number=".$account->number." account->cle=".$account->cle." account->iban=".$account->iban." country_code=".$country_code, LOG_DEBUG);
310 
311  if ($country_code == 'FR') { // France rules
312  $coef = array(62, 34, 3);
313  // Concatenation des differents codes.
314  $rib = strtolower(trim($account->code_banque).trim($account->code_guichet).trim($account->number).trim($account->cle));
315  // On remplace les eventuelles lettres par des chiffres.
316  //$rib = strtr($rib, "abcdefghijklmnopqrstuvwxyz","12345678912345678912345678"); //Ne marche pas
317  $rib = strtr($rib, "abcdefghijklmnopqrstuvwxyz", "12345678912345678923456789");
318  // Separation du rib en 3 groupes de 7 + 1 groupe de 2.
319  // Multiplication de chaque groupe par les coef du tableau
320 
321  for ($i = 0, $s = 0; $i < 3; $i++) {
322  $code = substr($rib, 7 * $i, 7);
323  $s += (0 + (int) $code) * $coef[$i];
324  }
325  // Soustraction du modulo 97 de $s a 97 pour obtenir la cle
326  $cle_rib = 97 - ($s % 97);
327  if ($cle_rib == $account->cle) {
328  return true;
329  }
330  return false;
331  }
332 
333  if ($country_code == 'BE') { // Belgium rules
334  }
335 
336  if ($country_code == 'ES') { // Spanish rules
337  $CCC = strtolower(trim($account->number));
338  $rib = strtolower(trim($account->code_banque).trim($account->code_guichet));
339  $cle_rib = strtolower(checkES($rib, $CCC));
340  if ($cle_rib == strtolower($account->cle)) {
341  return true;
342  }
343  return false;
344  }
345  if ($country_code == 'AU') { // Australian
346  if (strlen($account->code_banque) > 7) {
347  return false; // Sould be 6 but can be 123-456
348  } elseif (strlen($account->code_banque) < 6) {
349  return false; // Sould be 6
350  } else {
351  return true;
352  }
353  }
354 
355  // No particular rule
356  // If account is CompanyBankAccount class, we use number
357  // If account is Account class, we use num_compte
358  if (empty($account->number)) {
359  return false;
360  }
361 
362  return true;
363 }

374 function checkES($IentOfi, $InumCta)
375 {
376  if (empty($IentOfi) || empty($InumCta) || strlen($IentOfi) != 8 || strlen($InumCta) != 10) {
377  $keycontrol = "";
378  return $keycontrol;
379  }
380 
381  $ccc = $IentOfi.$InumCta;
382  $numbers = "1234567890";
383 
384  $i = 0;
385 
386  while ($i <= strlen($ccc) - 1) {
387  if (strpos($numbers, substr($ccc, $i, 1)) === false) {
388  $keycontrol = "";
389  return $keycontrol;
390  }
391  $i++;
392  }
393 
394  $values = array(1, 2, 4, 8, 5, 10, 9, 7, 3, 6);
395  $sum = 0;
396 
397  for ($i = 2; $i < 10; $i++) {
398  $sum += $values[$i] * substr($IentOfi, $i - 2, 1);
399  }
400 
401  $key = 11 - $sum % 11;
402 
403  if ($key == 10) {
404  $key = 1;
405  }
406  if ($key == 11) {
407  $key = 0;
408  }
409 
410  $keycontrol = $key;
411 
412  $sum = 0;
413 
414  for ($i = 0; $i < 11; $i++) {
415  $sum += $values[$i] * (int) substr($InumCta, $i, 1); //int to cast result of substr to a number
416  }
417 
418  $key = 11 - $sum % 11;
419 
420  if ($key == 10) {
421  $key = 1;
422  }
423  if ($key == 11) {
424  $key = 0;
425  }
426 
427  $keycontrol .= $key;
428  return $keycontrol;
429 }
