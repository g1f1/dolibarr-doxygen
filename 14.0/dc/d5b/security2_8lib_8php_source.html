321 function makesalt($type = CRYPT_SALT_LENGTH)
322 {
323  dol_syslog("makesalt type=".$type);
324  switch ($type) {
325  case 12: // 8 + 4
326  $saltlen = 8;
327  $saltprefix = '$1$';
328  $saltsuffix = '$';
329  break;
330  case 8: // 8 (Pour compatibilite, ne devrait pas etre utilise)
331  $saltlen = 8;
332  $saltprefix = '$1$';
333  $saltsuffix = '$';
334  break;
335  case 2: // 2
336  default: // by default, fall back on Standard DES (should work everywhere)
337  $saltlen = 2;
338  $saltprefix = '';
339  $saltsuffix = '';
340  break;
341  }
342  $salt = '';
343  while (dol_strlen($salt) < $saltlen) {
344  $salt .= chr(mt_rand(64, 126));
345  }
346 
347  $result = $saltprefix.$salt.$saltsuffix;
348  dol_syslog("makesalt return=".$result);
349  return $result;
350 }

358 function encodedecode_dbpassconf($level = 0)
359 {
360  dol_syslog("encodedecode_dbpassconf level=".$level, LOG_DEBUG);
361  $config = '';
362  $passwd = '';
363  $passwd_crypted = '';
364 
365  if ($fp = fopen(DOL_DOCUMENT_ROOT.'/conf/conf.php', 'r')) {
366  while (!feof($fp)) {
367  $buffer = fgets($fp, 4096);
368 
369  $lineofpass = 0;
370 
371  if (preg_match('/^[^#]*dolibarr_main_db_encrypted_pass[\s]*=[\s]*(.*)/i', $buffer, $reg)) { // Old way to save crypted value
372  $val = trim($reg[1]); // This also remove CR/LF
373  $val = preg_replace('/^["\']/', '', $val);
374  $val = preg_replace('/["\'][\s;]*$/', '', $val);
375  if (!empty($val)) {
376  $passwd_crypted = $val;

382  $val = trim($reg[1]); // This also remove CR/LF

385  if (preg_match('/crypted:/i', $buffer)) {
386  $val = preg_replace('/crypted:/i', '', $val);

388  $val = dol_decode($val);
389  $passwd = $val;

392  $val = dol_encode($val);

412  }
413  }

448 function getRandomPassword($generic = false, $replaceambiguouschars = null, $length = 32)
449 {
450  global $db, $conf, $langs, $user;
451 
452  $generated_password = '';
453  if ($generic) {
454  $lowercase = "qwertyuiopasdfghjklzxcvbnm";
455  $uppercase = "ASDFGHJKLZXCVBNMQWERTYUIOP";
456  $numbers = "1234567890";
457  $randomCode = "";
458  $nbofchar = round($length / 3);
459  $nbofcharlast = ($length - 2 * $nbofchar);
460  //var_dump($nbofchar.'-'.$nbofcharlast);
461  if (function_exists('random_int')) { // Cryptographic random
462  $max = strlen($lowercase) - 1;
463  for ($x = 0; $x < $nbofchar; $x++) {
464  $tmp = random_int(0, $max);
465  $randomCode .= $lowercase[$tmp];
466  }
467  $max = strlen($uppercase) - 1;
468  for ($x = 0; $x < $nbofchar; $x++) {
469  $tmp = random_int(0, $max);
470  $randomCode .= $uppercase[$tmp];
471  }
472  $max = strlen($numbers) - 1;
473  for ($x = 0; $x < $nbofcharlast; $x++) {
474  $tmp = random_int(0, $max);
475  $randomCode .= $numbers[$tmp];
476  }
477 
478  $generated_password = str_shuffle($randomCode);
479  } else {
480  // Old platform, non cryptographic random
481  $max = strlen($lowercase) - 1;
482  for ($x = 0; $x < $nbofchar; $x++) {
483  $tmp = mt_rand(0, $max);
484  $randomCode .= $lowercase[$tmp];
485  }
486  $max = strlen($uppercase) - 1;
487  for ($x = 0; $x < $nbofchar; $x++) {
488  $tmp = mt_rand(0, $max);
489  $randomCode .= $uppercase[$tmp];
490  }
491  $max = strlen($numbers) - 1;
492  for ($x = 0; $x < $nbofcharlast; $x++) {
493  $tmp = mt_rand(0, $max);
494  $randomCode .= $numbers[$tmp];
495  }
496 
497  $generated_password = str_shuffle($randomCode);
498  }
499  } elseif (!empty($conf->global->USER_PASSWORD_GENERATED)) {

512  $max = strlen($numbers) - 1;
