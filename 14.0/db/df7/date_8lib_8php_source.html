120 function dol_time_plus_duree($time, $duration_value, $duration_unit)
121 {
122  global $conf;
123 
124  if ($duration_value == 0) {
125  return $time;
126  }
127  if ($duration_unit == 'i') {
128  return $time + (60 * $duration_value);
129  }
130  if ($duration_unit == 'h') {
131  return $time + (3600 * $duration_value);
132  }
133  if ($duration_unit == 'w') {
134  return $time + (3600 * 24 * 7 * $duration_value);
135  }
136 
137  $deltastring = 'P';
138 
139  if ($duration_value > 0) {
140  $deltastring .= abs($duration_value);
141  $sub = false;
142  }
143  if ($duration_value < 0) {
144  $deltastring .= abs($duration_value);
145  $sub = true;
146  }
147  if ($duration_unit == 'd') {
148  $deltastring .= "D";
149  }
150  if ($duration_unit == 'm') {
151  $deltastring .= "M";
152  }
153  if ($duration_unit == 'y') {
154  $deltastring .= "Y";
155  }
156 
157  $date = new DateTime();
158  if (!empty($conf->global->MAIN_DATE_IN_MEMORY_ARE_GMT)) {
159  $date->setTimezone(new DateTimeZone('UTC'));
160  }
161  $date->setTimestamp($time);
162  $interval = new DateInterval($deltastring);
163 
164  if ($sub) {

166  } else {
167  $date->add($interval);
168  }

211 function convertSecondToTime($iSecond, $format = 'all', $lengthOfDay = 86400, $lengthOfWeek = 7)
212 {
213  global $langs;
214 
215  if (empty($lengthOfDay)) {
216  $lengthOfDay = 86400; // 1 day = 24 hours
217  }
218  if (empty($lengthOfWeek)) {
219  $lengthOfWeek = 7; // 1 week = 7 days
220  }
221 
222  if ($format == 'all' || $format == 'allwithouthour' || $format == 'allhour' || $format == 'allhourmin' || $format == 'allhourminsec') {
223  if ((int) $iSecond === 0) {
224  return '0'; // This is to avoid having 0 return a 12:00 AM for en_US
225  }
226 
227  $sTime = '';
228  $sDay = 0;
229  $sWeek = 0;
230 
231  if ($iSecond >= $lengthOfDay) {
232  for ($i = $iSecond; $i >= $lengthOfDay; $i -= $lengthOfDay) {
233  $sDay++;
234  $iSecond -= $lengthOfDay;
235  }
236  $dayTranslate = $langs->trans("Day");
237  if ($iSecond >= ($lengthOfDay * 2)) {
238  $dayTranslate = $langs->trans("Days");
239  }
240  }
241 
242  if ($lengthOfWeek < 7) {
243  if ($sDay) {
244  if ($sDay >= $lengthOfWeek) {
245  $sWeek = (int) (($sDay - $sDay % $lengthOfWeek) / $lengthOfWeek);
246  $sDay = $sDay % $lengthOfWeek;
247  $weekTranslate = $langs->trans("DurationWeek");
248  if ($sWeek >= 2) {
249  $weekTranslate = $langs->trans("DurationWeeks");
250  }
251  $sTime .= $sWeek.' '.$weekTranslate.' ';
252  }
253  }
254  }
255  if ($sDay > 0) {
256  $dayTranslate = $langs->trans("Day");
257  if ($sDay > 1) {
258  $dayTranslate = $langs->trans("Days");
259  }
260  $sTime .= $sDay.' '.strtolower(dol_substr($dayTranslate, 0, 1)).'. ';
261  }
262 
263  if ($format == 'all') {
264  if ($iSecond || empty($sDay)) {
265  $sTime .= dol_print_date($iSecond, 'hourduration', true);
266  }
267  } elseif ($format == 'allhourminsec') {
268  return sprintf("%02d", ($sWeek * $lengthOfWeek * 24 + $sDay * 24 + (int) floor($iSecond / 3600))).':'.sprintf("%02d", ((int) floor(($iSecond % 3600) / 60))).':'.sprintf("%02d", ((int) ($iSecond % 60)));
269  } elseif ($format == 'allhourmin') {
270  return sprintf("%02d", ($sWeek * $lengthOfWeek * 24 + $sDay * 24 + (int) floor($iSecond / 3600))).':'.sprintf("%02d", ((int) floor(($iSecond % 3600) / 60)));
271  } elseif ($format == 'allhour') {
272  return sprintf("%02d", ($sWeek * $lengthOfWeek * 24 + $sDay * 24 + (int) floor($iSecond / 3600)));
273  }
274  } elseif ($format == 'hour') { // only hour part
275  $sTime = dol_print_date($iSecond, '%H', true);
276  } elseif ($format == 'fullhour') {
277  if (!empty($iSecond)) {
278  $iSecond = $iSecond / 3600;
279  } else {
280  $iSecond = 0;
281  }
282  $sTime = $iSecond;
283  } elseif ($format == 'min') { // only min part
284  $sTime = dol_print_date($iSecond, '%M', true);
285  } elseif ($format == 'sec') { // only sec part
286  $sTime = dol_print_date($iSecond, '%S', true);
287  } elseif ($format == 'month') { // only month part
288  $sTime = dol_print_date($iSecond, '%m', true);
289  } elseif ($format == 'year') { // only year part
290  $sTime = dol_print_date($iSecond, '%Y', true);
291  }
292  return trim($sTime);
293 }

310 {

318  $sqldate .= ($excludefirstand ? "" : " AND ").$datefield." BETWEEN '".$db->idate(dol_mktime(0, 0, 0, $month_date, $day_date, $year_date, $gm));

354  if (preg_match('/^([0-9]+)\/([0-9]+)\/([0-9]+)\s?([0-9]+)?:?([0-9]+)?:?([0-9]+)?/i', $string, $reg)) {
