267  public function query($query, $usesavepoint = 0, $type = 'auto')
268  {
269  global $conf;
270 
271  $query = trim($query);
272 
273  if (!in_array($query, array('BEGIN', 'COMMIT', 'ROLLBACK'))) {
274  $SYSLOG_SQL_LIMIT = 10000; // limit log to 10kb per line to limit DOS attacks
275  dol_syslog('sql='.substr($query, 0, $SYSLOG_SQL_LIMIT), LOG_DEBUG);
276  }
277  if (empty($query)) {
278  return false; // Return false = error if empty request
279  }
280 
281  if (!$this->database_name) {
282  // Ordre SQL ne necessitant pas de connexion a une base (exemple: CREATE DATABASE)
283  $ret = $this->db->query($query);
284  } else {
285  $ret = $this->db->query($query);
286  }
287 
288  if (!preg_match("/^COMMIT/i", $query) && !preg_match("/^ROLLBACK/i", $query)) {
289  // Si requete utilisateur, on la sauvegarde ainsi que son resultset
290  if (!$ret) {
291  $this->lastqueryerror = $query;
292  $this->lasterror = $this->error();
293  $this->lasterrno = $this->errno();
294 
295  if ($conf->global->SYSLOG_LEVEL < LOG_DEBUG) {
296  dol_syslog(get_class($this)."::query SQL Error query: ".$query, LOG_ERR); // Log of request was not yet done previously
297  }
298  dol_syslog(get_class($this)."::query SQL Error message: ".$this->lasterrno." ".$this->lasterror, LOG_ERR);
299  //var_dump(debug_print_backtrace());
300  }
301  $this->lastquery = $query;
302  $this->_results = $ret;
303  }
304 
305  return $ret;
306  }

707  public function DDLCreateTable($table, $fields, $primary_key, $type, $unique_keys = null, $fulltext_keys = null, $keys = null)
708  {
709  // phpcs:enable
710  // FIXME: $fulltext_keys parameter is unused
711 
712  // cles recherchees dans le tableau des descriptions (fields) : type,value,attribute,null,default,extra
713  // ex. : $fields['rowid'] = array('type'=>'int','value'=>'11','null'=>'not null','extra'=> 'auto_increment');
714  $sql = "CREATE TABLE ".$table."(";
715  $i = 0;
716  $sqlfields = array();
717  foreach ($fields as $field_name => $field_desc) {
718  $sqlfields[$i] = $field_name." ";
719  $sqlfields[$i] .= $field_desc['type'];
720  if (preg_match("/^[^\s]/i", $field_desc['value'])) {
721  $sqlfields[$i] .= "(".$field_desc['value'].")";
722  }
723  if (preg_match("/^[^\s]/i", $field_desc['attribute'])) {
724  $sqlfields[$i] .= " ".$field_desc['attribute'];
725  }
726  if (preg_match("/^[^\s]/i", $field_desc['default'])) {
727  if ((preg_match("/null/i", $field_desc['default'])) || (preg_match("/CURRENT_TIMESTAMP/i", $field_desc['default']))) {
728  $sqlfields[$i] .= " default ".$field_desc['default'];
729  } else {
730  $sqlfields[$i] .= " default '".$this->escape($field_desc['default'])."'";
731  }
732  }
733  if (preg_match("/^[^\s]/i", $field_desc['null'])) {
734  $sqlfields[$i] .= " ".$field_desc['null'];
735  }
736  if (preg_match("/^[^\s]/i", $field_desc['extra'])) {
737  $sqlfields[$i] .= " ".$field_desc['extra'];
738  }
739  $i++;
740  }
741  if ($primary_key != "") {
742  $pk = "primary key(".$primary_key.")";
743  }
744 
745  if (is_array($unique_keys)) {
746  $i = 0;
747  foreach ($unique_keys as $key => $value) {
748  $sqluq[$i] = "UNIQUE KEY '".$key."' ('".$this->escape($value)."')";
749  $i++;
750  }
751  }
752  if (is_array($keys)) {
753  $i = 0;
754  foreach ($keys as $key => $value) {
755  $sqlk[$i] = "KEY ".$key." (".$value.")";
756  $i++;
757  }
758  }
759  $sql .= implode(',', $sqlfields);
760  if ($primary_key != "") {
761  $sql .= ",".$pk;
762  }
763  if ($unique_keys != "") {
764  $sql .= ",".implode(',', $sqluq);
765  }
766  if (is_array($keys)) {
767  $sql .= ",".implode(',', $sqlk);
768  }
769  $sql .= ") engine=".$type;
770 
771  if (!$this->query($sql)) {
772  return -1;
773  } else {
774  return 1;
775  }
776  }
