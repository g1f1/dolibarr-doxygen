61 function realCharForNumericEntities($matches)
62 {
63  $newstringnumentity = preg_replace('/;$/', '', $matches[1]);
64  //print ' $newstringnumentity='.$newstringnumentity;
65 
66  if (preg_match('/^x/i', $newstringnumentity)) {
67  $newstringnumentity = hexdec(preg_replace('/^x/i', '', $newstringnumentity));
68  }
69 
70  // The numeric value we don't want as entities because they encode ascii char, and why using html entities on ascii except for haking ?
71  if (($newstringnumentity >= 65 && $newstringnumentity <= 90) || ($newstringnumentity >= 97 && $newstringnumentity <= 122)) {
72  return chr((int) $newstringnumentity);
73  }
74 
75  return '&#'.$matches[1]; // Value will be unchanged because regex was /&#( )/
76 }

87 function testSqlAndScriptInject($val, $type)
88 {
89  // Decode string first because a lot of things are obfuscated by encoding or multiple encoding.
90  // So <svg o&#110;load='console.log(&quot;123&quot;)' become <svg onload='console.log(&quot;123&quot;)'
91  // So "&colon;&apos;" become ":'" (due to ENT_HTML5)
92  // Loop to decode until no more thing to decode.
93  //print "before decoding $val\n";
94  do {
95  $oldval = $val;
96  $val = html_entity_decode($val, ENT_QUOTES | ENT_HTML5);
97  $val = preg_replace_callback('/&#(x?[0-9][0-9a-f]+)/i', 'realCharForNumericEntities', $val); // Sometimes we have entities without the ; at end so html_entity_decode does not work but entities is still interpreted by browser.
98  } while ($oldval != $val);
99  //print "after decoding $val\n";
100 
101  // We clean string because some hacks try to obfuscate evil strings by inserting non printable chars. Example: 'java(ascci09)scr(ascii00)ipt' is processed like 'javascript' (whatever is place of evil ascii char)
102  // We should use dol_string_nounprintableascii but function is not yet loaded/available
103  $val = preg_replace('/[\x00-\x1F\x7F]/u', '', $val); // /u operator makes UTF8 valid characters being ignored so are not included into the replace
104  // We clean html comments because some hacks try to obfuscate evil strings by inserting HTML comments. Example: on<!-- -->error=alert(1)
105  $val = preg_replace('/<!--[^>]*-->/', '', $val);
106 
107  $inj = 0;
108  // For SQL Injection (only GET are used to scan for such injection strings)
109  if ($type == 1 || $type == 3) {
110  $inj += preg_match('/delete\s+from/i', $val);
111  $inj += preg_match('/create\s+table/i', $val);
112  $inj += preg_match('/insert\s+into/i', $val);
113  $inj += preg_match('/select\s+from/i', $val);
114  $inj += preg_match('/into\s+(outfile|dumpfile)/i', $val);
115  $inj += preg_match('/user\s*\(/i', $val); // avoid to use function user() or mysql_user() that return current database login
116  $inj += preg_match('/information_schema/i', $val); // avoid to use request that read information_schema database
117  $inj += preg_match('/<svg/i', $val); // <svg can be allowed in POST
118  $inj += preg_match('/update.+set.+=/i', $val);
119  $inj += preg_match('/union.+select/i', $val);
120  }
121  if ($type == 3) {
122  $inj += preg_match('/select|update|delete|truncate|replace|group\s+by|concat|count|from|union/i', $val);
123  }
124  if ($type != 2) { // Not common key strings, so we can check them both on GET and POST
125  $inj += preg_match('/updatexml\(/i', $val);
126  $inj += preg_match('/(\.\.%2f)+/i', $val);
127  $inj += preg_match('/\s@@/', $val);
128  }
129  // For XSS Injection done by closing textarea to execute content into a textarea field
130  $inj += preg_match('/<\/textarea/i', $val);
131  // For XSS Injection done by adding javascript with script
132  // This is all cases a browser consider text is javascript:
133  // When it found '<script', 'javascript:', '<style', 'onload\s=' on body tag, '="&' on a tag size with old browsers
134  // All examples on page: http://ha.ckers.org/xss.html#XSScalc
135  // More on https://www.owasp.org/index.php/XSS_Filter_Evasion_Cheat_Sheet
136  $inj += preg_match('/<audio/i', $val);
137  $inj += preg_match('/<embed/i', $val);
138  $inj += preg_match('/<iframe/i', $val);
139  $inj += preg_match('/<object/i', $val);
140  $inj += preg_match('/<script/i', $val);
141  $inj += preg_match('/Set\.constructor/i', $val); // ECMA script 6
142  if (!defined('NOSTYLECHECK')) {
143  $inj += preg_match('/<style/i', $val);
144  }
145  $inj += preg_match('/base\s+href/si', $val);
146  $inj += preg_match('/=data:/si', $val);
147  // List of dom events is on https://www.w3schools.com/jsref/dom_obj_event.asp and https://developer.mozilla.org/en-US/docs/Web/API/GlobalEventHandlers
148  $inj += preg_match('/on(mouse|drag|key|load|touch|pointer|select|transition)([a-z]*)\s*=/i', $val); // onmousexxx can be set on img or any html tag like <img title='...' onmouseover=alert(1)>
149  $inj += preg_match('/on(abort|afterprint|animation|auxclick|beforecopy|beforecut|beforeprint|beforeunload|blur|cancel|canplay|canplaythrough|change|click|close|contextmenu|cuechange|copy|cut)\s*=/i', $val);
150  $inj += preg_match('/on(dblclick|drop|durationchange|emptied|ended|error|focus|focusin|focusout|formdata|gotpointercapture|hashchange|input|invalid)\s*=/i', $val);
151  $inj += preg_match('/on(lostpointercapture|offline|online|pagehide|pageshow)\s*=/i', $val);
152  $inj += preg_match('/on(paste|pause|play|playing|progress|ratechange|reset|resize|scroll|search|seeked|seeking|show|stalled|start|submit|suspend)\s*=/i', $val);
153  $inj += preg_match('/on(timeupdate|toggle|unload|volumechange|waiting|wheel)\s*=/i', $val);
154 
155  // We refuse html into html because some hacks try to obfuscate evil strings by inserting HTML into HTML. Example: <img on<a>error=alert(1) to bypass test on onerror
156  $tmpval = preg_replace('/<[^<]+>/', '', $val);
157  // List of dom events is on https://www.w3schools.com/jsref/dom_obj_event.asp and https://developer.mozilla.org/en-US/docs/Web/API/GlobalEventHandlers
158  $inj += preg_match('/on(mouse|drag|key|load|touch|pointer|select|transition)([a-z]*)\s*=/i', $val); // onmousexxx can be set on img or any html tag like <img title='...' onmouseover=alert(1)>
159  $inj += preg_match('/on(abort|afterprint|animation|auxclick|beforeprint|beforeunload|blur|cancel|canplay|canplaythrough|change|click|close|contextmenu|cuechange|copy|cut)\s*=/i', $tmpval);
160  $inj += preg_match('/on(dblclick|drop|durationchange|emptied|ended|error|focus|focusin|focusout|formdata|gotpointercapture|hashchange|input|invalid)\s*=/i', $tmpval);
161  $inj += preg_match('/on(lostpointercapture|offline|online|pagehide|pageshow)\s*=/i', $tmpval);
162  $inj += preg_match('/on(paste|pause|play|playing|progress|ratechange|reset|resize|scroll|search|seeked|seeking|show|stalled|start|submit|suspend)\s*=/i', $tmpval);
163  $inj += preg_match('/on(timeupdate|toggle|unload|volumechange|waiting|wheel)\s*=/i', $tmpval);
164 
165  //$inj += preg_match('/on[A-Z][a-z]+\*=/', $val); // To lock event handlers onAbort(), ...
166  $inj += preg_match('/&#58;|&#0000058|&#x3A/i', $val); // refused string ':' encoded (no reason to have it encoded) to lock 'javascript:...'
167 
168  $inj += preg_match('/javascript\s*:/i', $val);
169  $inj += preg_match('/vbscript\s*:/i', $val);
170  // For XSS Injection done by adding javascript closing html tags like with onmousemove, etc... (closing a src or href tag with not cleaned param)
171  if ($type == 1) {
172  $val = str_replace('enclosure="', 'enclosure=X', $val); // We accept enclosure=" for the export/import module
173  $inj += preg_match('/"/i', $val); // We refused " in GET parameters value.
174  }
175  if ($type == 2) {
176  $inj += preg_match('/[:;"\'<>\?\(\){}\$%]/', $val); // PHP_SELF is a file system (or url path without parameters). It can contains spaces.
177  }
178 
179  return $inj;
180 }
