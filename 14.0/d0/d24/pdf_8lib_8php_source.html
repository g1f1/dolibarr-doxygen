336 function pdfGetHeightForHtmlContent(&$pdf, $htmlcontent)
337 {
338  // store current object
339  $pdf->startTransaction();
340  // store starting values
341  $start_y = $pdf->GetY();
342  //var_dump($start_y);
343  $start_page = $pdf->getPage();
344  // call printing functions with content
345  $pdf->writeHTMLCell(0, 0, 0, $start_y, $htmlcontent, 0, 1, false, true, 'J', true);
346  // get the new Y
347  $end_y = $pdf->GetY();
348  $end_page = $pdf->getPage();
349  // calculate height
350  $height = 0;
351  if ($end_page == $start_page) {
352  $height = $end_y - $start_y;
353  } else {
354  for ($page = $start_page; $page <= $end_page; ++$page) {
355  $pdf->setPage($page);
356  $tmpm = $pdf->getMargins();
357  $tMargin = $tmpm['top'];
358  if ($page == $start_page) {
359  // first page
360  $height = $pdf->getPageHeight() - $start_y - $pdf->getBreakMargin();
361  } elseif ($page == $end_page) {
362  // last page
363  $height = $end_y - $tMargin;
364  } else {
365  $height = $pdf->getPageHeight() - $tMargin - $pdf->getBreakMargin();
366  }
367  }
368  }
369  // restore previous object
370  $pdf = $pdf->rollbackTransaction();
371 
372  return $height;
373 }
