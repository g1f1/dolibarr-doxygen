117 function getMarginInfos($pvht, $remise_percent, $tva_tx, $localtax1_tx, $localtax2_tx, $fk_pa, $paht)
118 {
119  global $db, $conf;
120 
121  $marge_tx_ret = '';
122  $marque_tx_ret = '';
123 
124  if ($fk_pa > 0 && empty($paht)) {
125  require_once DOL_DOCUMENT_ROOT.'/fourn/class/fournisseur.product.class.php';
126  $product = new ProductFournisseur($db);
127  if ($product->fetch_product_fournisseur_price($fk_pa)) {
128  $paht_ret = $product->fourn_unitprice * (1 - $product->fourn_remise_percent / 100);
129  } else {
130  $paht_ret = $paht;
131  }
132  } else {
133  $paht_ret = $paht;
134  }
135 
136  // Calculate selling unit price including line discount
137  // We don't use calculate_price, because this function is dedicated to calculation of total with accuracy of total. We need an accuracy of a unit price.
138  // Also we must not apply rounding on non decimal rule defined by option MAIN_ROUNDING_RULE_TOT
139  $pu_ht_remise = $pvht * (1 - ($remise_percent / 100));
140  $pu_ht_remise = price2num($pu_ht_remise, 'MU');
141 
142  // calcul marge
143  if ($pu_ht_remise < 0) {
144  $marge = -1 * (abs($pu_ht_remise) - $paht_ret);
145  } else {
146  $marge = $pu_ht_remise - $paht_ret;
147  }
148 
149  // calcul taux marge
150  if ($paht_ret != 0) {
151  $marge_tx_ret = (100 * $marge) / $paht_ret;
152  }
153  // calcul taux marque
154  if ($pu_ht_remise != 0) {
155  $marque_tx_ret = (100 * $marge) / $pu_ht_remise;
156  }
157 
158  return array($paht_ret, $marge_tx_ret, $marque_tx_ret);
159 }
