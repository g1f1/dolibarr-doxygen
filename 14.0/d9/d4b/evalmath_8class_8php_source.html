145  public function evaluate($expr)
146  {
147  $this->last_error = null;
148  $this->last_error_code = null;
149  $expr = trim($expr);
150  if (substr($expr, - 1, 1) == ';') {
151  $expr = substr($expr, 0, strlen($expr) - 1); // strip semicolons at the end
152  }
153  // ===============
154  // is it a variable assignment?
155  $matches = array();
156  if (preg_match('/^\s*([a-z]\w*)\s*=\s*(.+)$/', $expr, $matches)) {
157  if (in_array($matches[1], $this->vb)) { // make sure we're not assigning to a constant
158  return $this->trigger(1, "cannot assign to constant '$matches[1]'", $matches[1]);
159  }
160  if (($tmp = $this->pfx($this->nfx($matches[2]))) === false) {
161  return false; // get the result and make sure it's good
162  }
163  $this->v[$matches[1]] = $tmp; // if so, stick it in the variable array
164  return $this->v[$matches[1]]; // and return the resulting value
165  // ===============
166  // is it a function assignment?
167  } elseif (preg_match('/^\s*([a-z]\w*)\s*\(\s*([a-z]\w*(?:\s*,\s*[a-z]\w*)*)\s*\)\s*=\s*(.+)$/', $expr, $matches)) {
168  $fnn = $matches[1]; // get the function name
169  if (in_array($matches[1], $this->fb)) { // make sure it isn't built in
170  return $this->trigger(2, "cannot redefine built-in function '$matches[1]()'", $matches[1]);
171  }
172  $args = explode(",", preg_replace("/\s+/", "", $matches[2])); // get the arguments
173  if (($stack = $this->nfx($matches[3])) === false) {
174  return false; // see if it can be converted to postfix
175  }
176  $nbstack = count($stack);
177  for ($i = 0; $i < $nbstack; $i++) { // freeze the state of the non-argument variables
178  $token = $stack[$i];
179  if (preg_match('/^[a-z]\w*$/', $token) and !in_array($token, $args)) {
180  if (array_key_exists($token, $this->v)) {
181  $stack[$i] = $this->v[$token];
182  } else {
183  return $this->trigger(3, "undefined variable '$token' in function definition", $token);
184  }
185  }
186  }
187  $this->f[$fnn] = array('args' => $args, 'func' => $stack);
188  return true;
189  // ===============
190  } else {
191  return $this->pfx($this->nfx($expr)); // straight up evaluation, woo
192  }
193  }

230  private function nfx($expr)
231  {
232  $index = 0;
233  $stack = new EvalMathStack();
234  $output = array(); // postfix form of expression, to be passed to pfx()
235  $expr = trim(strtolower($expr));
236 
237  $ops = array('+', '-', '*', '/', '^', '_');
238  $ops_r = array('+' => 0, '-' => 0, '*' => 0, '/' => 0, '^' => 1); // right-associative operator?
239  $ops_p = array('+' => 0, '-' => 0, '*' => 1, '/' => 1, '_' => 1, '^' => 2); // operator precedence
240 
241  $expecting_op = false; // we use this in syntax-checking the expression
242  // and determining when a - is a negation
243 
244  $matches = array();
245  if (preg_match("/[^\w\s+*^\/()\.,-]/", $expr, $matches)) { // make sure the characters are all good
246  return $this->trigger(4, "illegal character '{$matches[0]}'", $matches[0]);
247  }
248 
249  while (1) { // 1 Infinite Loop ;)
250  $op = substr($expr, $index, 1); // get the first character at the current index
251  // find out if we're currently at the beginning of a number/variable/function/parenthesis/operand
252  $match = array();
253  $ex = preg_match('/^([a-z]\w*\(?|\d+(?:\.\d*)?|\.\d+|\()/', substr($expr, $index), $match);
254  // ===============
255  if ($op == '-' and !$expecting_op) { // is it a negation instead of a minus?
256  $stack->push('_'); // put a negation on the stack
257  $index++;
258  } elseif ($op == '_') { // we have to explicitly deny this, because it's legal on the stack
259  return $this->trigger(4, "illegal character '_'", "_"); // but not in the input expression
260  // ===============
261  } elseif ((in_array($op, $ops) or $ex) and $expecting_op) { // are we putting an operator on the stack?
262  if ($ex) { // are we expecting an operator but have a number/variable/function/opening parethesis?
263  $op = '*';
264  $index--; // it's an implicit multiplication
265  }
266  // heart of the algorithm:
267  while ($stack->count > 0 and ($o2 = $stack->last()) and in_array($o2, $ops) and ($ops_r[$op] ? $ops_p[$op] < $ops_p[$o2] : $ops_p[$op] <= $ops_p[$o2])) {
268  $output[] = $stack->pop(); // pop stuff off the stack into the output
269  }
270  // many thanks: http://en.wikipedia.org/wiki/Reverse_Polish_notation#The_algorithm_in_detail
271  $stack->push($op); // finally put OUR operator onto the stack
272  $index++;
273  $expecting_op = false;
274  // ===============
275  } elseif ($op == ')' and $expecting_op) { // ready to close a parenthesis?
276  while (($o2 = $stack->pop()) != '(') { // pop off the stack back to the last (
277  if (is_null($o2)) {
278  return $this->trigger(5, "unexpected ')'", ")");
279  } else {
280  $output[] = $o2;
281  }
282  }
283  if (preg_match("/^([a-z]\w*)\($/", $stack->last(2), $matches)) { // did we just close a function?
284  $fnn = $matches[1]; // get the function name
285  $arg_count = $stack->pop(); // see how many arguments there were (cleverly stored on the stack, thank you)
286  $output[] = $stack->pop(); // pop the function and push onto the output
287  if (in_array($fnn, $this->fb)) { // check the argument count
288  if ($arg_count > 1) {
289  return $this->trigger(6, "wrong number of arguments ($arg_count given, 1 expected)", array($arg_count, 1));
290  }
291  } elseif (array_key_exists($fnn, $this->f)) {
292  if ($arg_count != count($this->f[$fnn]['args'])) {
293  return $this->trigger(6, "wrong number of arguments ($arg_count given, ".count($this->f[$fnn]['args'])." expected)", array($arg_count, count($this->f[$fnn]['args'])));
294  }
295  } else { // did we somehow push a non-function on the stack? this should never happen
296  return $this->trigger(7, "internal error");
297  }
298  }
299  $index++;
300  // ===============
301  } elseif ($op == ',' and $expecting_op) { // did we just finish a function argument?
302  while (($o2 = $stack->pop()) != '(') {
303  if (is_null($o2)) {
304  return $this->trigger(5, "unexpected ','", ","); // oops, never had a (
305  } else {
306  $output[] = $o2; // pop the argument expression stuff and push onto the output
307  }
308  }
309  // make sure there was a function
310  if (!preg_match("/^([a-z]\w*)\($/", $stack->last(2), $matches)) {
311  return $this->trigger(5, "unexpected ','", ",");
312  }
313  $stack->push($stack->pop() + 1); // increment the argument count
314  $stack->push('('); // put the ( back on, we'll need to pop back to it again
315  $index++;
316  $expecting_op = false;
317  // ===============
318  } elseif ($op == '(' and !$expecting_op) {
319  $stack->push('('); // that was easy
320  $index++;
321  $allow_neg = true;
322  // ===============
323  } elseif ($ex and !$expecting_op) { // do we now have a function/variable/number?
324  $expecting_op = true;
325  $val = $match[1];
326  if (preg_match("/^([a-z]\w*)\($/", $val, $matches)) { // may be func, or variable w/ implicit multiplication against parentheses...
327  if (in_array($matches[1], $this->fb) or array_key_exists($matches[1], $this->f)) { // it's a func
328  $stack->push($val);
329  $stack->push(1);
330  $stack->push('(');
331  $expecting_op = false;
332  } else { // it's a var w/ implicit multiplication
333  $val = $matches[1];
334  $output[] = $val;
335  }
336  } else { // it's a plain old var or num
337  $output[] = $val;
338  }
339  $index += strlen($val);
340  // ===============
341  } elseif ($op == ')') { // miscellaneous error checking
342  return $this->trigger(5, "unexpected ')'", ")");
343  } elseif (in_array($op, $ops) and !$expecting_op) {
344  return $this->trigger(8, "unexpected operator '$op'", $op);
345  } else { // I don't even want to know what you did to get here
346  return $this->trigger(9, "an unexpected error occured");
347  }
348  if ($index == strlen($expr)) {
349  if (in_array($op, $ops)) { // did we end with an operator? bad.
350  return $this->trigger(10, "operator '$op' lacks operand", $op);
351  } else {
352  break;
353  }
354  }
355  while (substr($expr, $index, 1) == ' ') { // step the index past whitespace (pretty much turns whitespace
356  $index++; // into implicit multiplication if no operator is there)
357  }
358  }
359  while (!is_null($op = $stack->pop())) { // pop everything off the stack and push onto output
360  if ($op == '(') {
361  return $this->trigger(11, "expecting ')'", ")"); // if there are (s on the stack, ()s were unbalanced
362  }
363  $output[] = $op;
364  }
365  return $output;
366  }

375  private function pfx($tokens, $vars = array())
376  {
377  if ($tokens == false) {
378  return false;
379  }
380 
381  $stack = new EvalMathStack();
382 
383  foreach ($tokens as $token) { // nice and easy
384  // if the token is a binary operator, pop two values off the stack, do the operation, and push the result back on
385  $matches = array();
386  if (in_array($token, array('+', '-', '*', '/', '^'))) {
387  if (is_null($op2 = $stack->pop())) {
388  return $this->trigger(12, "internal error");
389  }
390  if (is_null($op1 = $stack->pop())) {
391  return $this->trigger(13, "internal error");
392  }
393  switch ($token) {
394  case '+':
395  $stack->push($op1 + $op2);
396  break;
397  case '-':
398  $stack->push($op1 - $op2);
399  break;
400  case '*':
401  $stack->push($op1 * $op2);
402  break;
403  case '/':
404  if ($op2 == 0) {
405  return $this->trigger(14, "division by zero");
406  }
407  $stack->push($op1 / $op2);
408  break;
409  case '^':
410  $stack->push(pow($op1, $op2));
411  break;
412  }
413  // if the token is a unary operator, pop one value off the stack, do the operation, and push it back on
414  } elseif ($token == "_") {
415  $stack->push(-1 * $stack->pop());
416  // if the token is a function, pop arguments off the stack, hand them to the function, and push the result back on
417  } elseif (preg_match("/^([a-z]\w*)\($/", $token, $matches)) { // it's a function!
418  $fnn = $matches[1];
419  if (in_array($fnn, $this->fb)) { // built-in function:
420  if (is_null($op1 = $stack->pop())) {
421  return $this->trigger(15, "internal error");
422  }
423  $fnn = preg_replace("/^arc/", "a", $fnn); // for the 'arc' trig synonyms
424  if ($fnn == 'ln') {
425  $fnn = 'log';
426  }
427  eval('$stack->push('.$fnn.'($op1));'); // perfectly safe eval()
428  } elseif (array_key_exists($fnn, $this->f)) { // user function
429  // get args
430  $args = array();
431  for ($i = count($this->f[$fnn]['args']) - 1; $i >= 0; $i--) {
432  if (is_null($args[$this->f[$fnn]['args'][$i]] = $stack->pop())) {
433  return $this->trigger(16, "internal error");
434  }
435  }
436  $stack->push($this->pfx($this->f[$fnn]['func'], $args)); // yay... recursion!!!!
437  }
438  // if the token is a number or variable, push it on the stack
439  } else {
440  if (is_numeric($token)) {
441  $stack->push($token);
442  } elseif (array_key_exists($token, $this->v)) {
443  $stack->push($this->v[$token]);
444  } elseif (array_key_exists($token, $vars)) {
445  $stack->push($vars[$token]);
446  } else {
447  return $this->trigger(17, "undefined variable '$token'", $token);
448  }
449  }
450  }
451  // when we're out of tokens, the stack should have a single element, the final result
452  if ($stack->count != 1) {
453  return $this->trigger(18, "internal error");
454  }
455  return $stack->pop();
456  }
