590  public function calculate_date_lim_reglement($cond_reglement = 0)

617  $obj = $this->db->fetch_object($resqltemp);
618  $cdr_nbjour = $obj->nbjour;
619  $cdr_type = $obj->type_cdr;
620  $cdr_decalage = $obj->decalage;
621  }

623  $this->error = $this->db->error();

628  /* Definition de la date limite */
629

631  if ($cdr_type == 0) {
632  $datelim = $this->date + ($cdr_nbjour * 3600 * 24);
633 
634  $datelim += ($cdr_decalage * 3600 * 24);
635  } elseif ($cdr_type == 1) {
636  // 1 : application of the "end of the month" rule
637  $datelim = $this->date + ($cdr_nbjour * 3600 * 24);
638

640  $annee = date('Y', $datelim);
641  if ($mois == 12) {
642  $mois = 1;

644  } else {
645  $mois += 1;
646  }
647  // We move at the beginning of the next month, and we take a day off
648  $datelim = dol_mktime(12, 0, 0, $mois, 1, $annee);
649  $datelim -= (3600 * 24);
650 
651  $datelim += ($cdr_decalage * 3600 * 24);
652  } elseif ($cdr_type == 2 && !empty($cdr_decalage)) {
653  // 2 : application of the rule, the N of the current or next month
654  include_once DOL_DOCUMENT_ROOT.'/core/lib/date.lib.php';
655  $datelim = $this->date + ($cdr_nbjour * 3600 * 24);
656 
657  $date_piece = dol_mktime(0, 0, 0, date('m', $datelim), date('d', $datelim), date('Y', $datelim)); // Sans les heures minutes et secondes
658  $date_lim_current = dol_mktime(0, 0, 0, date('m', $datelim), $cdr_decalage, date('Y', $datelim)); // Sans les heures minutes et secondes
659  $date_lim_next = dol_time_plus_duree($date_lim_current, 1, 'm'); // Add 1 month
660 
661  $diff = $date_piece - $date_lim_current;
662 
663  if ($diff < 0) {
664  $datelim = $date_lim_current;
665  } else {
666  $datelim = $date_lim_next;
667  }
668  } else {
669  return 'Bad value for type_cdr in database for record cond_reglement = '.$cond_reglement;
670  }
671 
672  return $datelim;
673  }
