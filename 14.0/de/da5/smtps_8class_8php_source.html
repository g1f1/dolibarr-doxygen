374  $_retVal = true;
375 
376  // We have to make sure the HOST given is valid
377  // This is done here because '@fsockopen' will not give me this
378  // information if it failes to connect because it can't find the HOST
379  $host = $this->getHost();
380  $usetls = preg_match('@tls://@i', $host);
381 
382  $host = preg_replace('@tcp://@i', '', $host); // Remove prefix
383  $host = preg_replace('@ssl://@i', '', $host); // Remove prefix
384  $host = preg_replace('@tls://@i', '', $host); // Remove prefix
385 
386  // @CHANGE LDR
387  include_once DOL_DOCUMENT_ROOT.'/core/lib/functions2.lib.php';
388

390  $this->_setErr(99, $host.' is either offline or is an invalid host name.');
391  $_retVal = false;
392  } else {
393  if (function_exists('stream_socket_client') && !empty($this->_options)) {
394  $socket_context = stream_context_create($this->_options); // An array of options for stream_context_create()

405  $this->socket = @fsockopen(

448  private function _server_authenticate()
449  {
450  // phpcs:enable
451  global $conf;
452 
453  // Send the RFC2554 specified EHLO.
454  // This improvment as provided by 'SirSir' to
455  // accomodate both SMTP AND ESMTP capable servers
456  $host = $this->getHost();
457  $usetls = preg_match('@tls://@i', $host);
458 
459  $host = preg_replace('@tcp://@i', '', $host); // Remove prefix
460  $host = preg_replace('@ssl://@i', '', $host); // Remove prefix
461  $host = preg_replace('@tls://@i', '', $host); // Remove prefix
462 
463  if ($usetls && !empty($conf->global->MAIN_SMTPS_ADD_TLS_TO_HOST_FOR_HELO)) {
464  $host = 'tls://'.$host;
465  }
466 
467  $hosth = $host;
468 
469  if (!empty($conf->global->MAIL_SMTP_USE_FROM_FOR_HELO)) {
470  // If the from to is 'aaa <bbb@ccc.com>', we will keep 'ccc.com'
471  $hosth = $this->getFrom('addr');
472  $hosth = preg_replace('/^.*</', '', $hosth);
473  $hosth = preg_replace('/>.*$/', '', $hosth);
474  $hosth = preg_replace('/.*@/', '', $hosth);
475  }
476 
477  if ($_retVal = $this->socket_send_str('EHLO '.$hosth, '250')) {
478  if ($usetls) {
479  /*
480  The following dialog illustrates how a client and server can start a TLS STARTTLS session
481  S: <waits for connection on TCP port 25>
482  C: <opens connection>
483  S: 220 mail.imc.org SMTP service ready
484  C: EHLO mail.ietf.org
485  S: 250-mail.imc.org offers a warm hug of welcome
486  S: 250 STARTTLS
487  C: STARTTLS
488  S: 220 Go ahead
489  C: <starts TLS negotiation>
490  C & S: <negotiate a TLS session>
491  C & S: <check result of negotiation>
492  // Second pass EHLO
493  C: EHLO client-domain.com
494  S: 250-server-domain.com
495  S: 250 AUTH LOGIN
496  C: <continues by sending an SMTP command
497  */
498  if (!$_retVal = $this->socket_send_str('STARTTLS', 220)) {
499  $this->_setErr(131, 'STARTTLS connection is not supported.');
500  return $_retVal;
501  }
502 
503  // Before 5.6.7:
504  // STREAM_CRYPTO_METHOD_SSLv23_CLIENT = STREAM_CRYPTO_METHOD_SSLv2_CLIENT|STREAM_CRYPTO_METHOD_SSLv3_CLIENT
505  // STREAM_CRYPTO_METHOD_TLS_CLIENT = STREAM_CRYPTO_METHOD_TLSv1_0_CLIENT|STREAM_CRYPTO_METHOD_TLSv1_1_CLIENT|STREAM_CRYPTO_METHOD_TLSv1_2_CLIENT
506  // PHP >= 5.6.7:
507  // STREAM_CRYPTO_METHOD_SSLv23_CLIENT = STREAM_CRYPTO_METHOD_TLSv1_0_CLIENT|STREAM_CRYPTO_METHOD_TLSv1_1_CLIENT|STREAM_CRYPTO_METHOD_TLSv1_2_CLIENT
508  // STREAM_CRYPTO_METHOD_TLS_CLIENT = STREAM_CRYPTO_METHOD_TLSv1_0_CLIENT
509 
510  $crypto_method = STREAM_CRYPTO_METHOD_TLS_CLIENT;
511  if (defined('STREAM_CRYPTO_METHOD_TLSv1_2_CLIENT')) {
512  $crypto_method |= STREAM_CRYPTO_METHOD_TLSv1_2_CLIENT;
513  $crypto_method |= STREAM_CRYPTO_METHOD_TLSv1_1_CLIENT;
514  }
515 
516  if (!stream_socket_enable_crypto($this->socket, true, $crypto_method)) {
517  $this->_setErr(132, 'STARTTLS connection failed.');
518  return $_retVal;
519  }
520  // Most server servers expect a 2nd pass of EHLO after TLS is established to get another time
521  // the answer with list of supported AUTH methods. They may differs between non STARTTLS and with STARTTLS.
522  if (!$_retVal = $this->socket_send_str('EHLO '.$hosth, '250')) {
523  $this->_setErr(126, '"'.$hosth.'" does not support authenticated connections.');
524  return $_retVal;
525  }
526  }
527 
528  // Default authentication method is LOGIN
529  if (empty($conf->global->MAIL_SMTP_AUTH_TYPE)) {
530  $conf->global->MAIL_SMTP_AUTH_TYPE = 'LOGIN';
531  }
532 
533  // Send Authentication to Server
534  // Check for errors along the way
535  switch ($conf->global->MAIL_SMTP_AUTH_TYPE) {
536  case 'NONE':
537  // Do not send the 'AUTH type' message. For test purpose, if you don't need authentication, it is better to not enter login/pass into setup.
538  $_retVal = true;
539  break;
540  case 'PLAIN':
541  $this->socket_send_str('AUTH PLAIN', '334');
542  // The error here just means the ID/password combo doesn't work.
543  $_retVal = $this->socket_send_str(base64_encode("\0".$this->_smtpsID."\0".$this->_smtpsPW), '235');
544  break;
545  case 'LOGIN': // most common case
546  default:
547  $_retVal = $this->socket_send_str('AUTH LOGIN', '334');
548  if (!$_retVal) {
549  $this->_setErr(130, 'Error when asking for AUTH LOGIN');
550  } else {
551  // User name will not return any error, server will take anything we give it.
552  $this->socket_send_str(base64_encode($this->_smtpsID), '334');
553  // The error here just means the ID/password combo doesn't work.
554  // There is not a method to determine which is the problem, ID or password
555  $_retVal = $this->socket_send_str(base64_encode($this->_smtpsPW), '235');
556  }
557  break;
558  }
559  if (!$_retVal) {
560  $this->_setErr(130, 'Invalid Authentication Credentials.');
561  }
562  } else {
563  $this->_setErr(126, '"'.$host.'" does not support authenticated connections.');
564  }
565 
566  return $_retVal;
567  }

574  public function sendMsg()
575  {
576  global $conf;
577 
581  $_retVal = false;
582 
583  // Connect to Server
584  if ($this->socket = $this->_server_connect()) {
585  // If a User ID *and* a password is given, assume Authentication is desired
586  if (!empty($this->_smtpsID) && !empty($this->_smtpsPW)) {
587  // Send the RFC2554 specified EHLO.
588  $_retVal = $this->_server_authenticate();
589  } else {
590  // This is a "normal" SMTP Server "handshack"
591  // Send the RFC821 specified HELO.
592  $host = $this->getHost();
593  $usetls = preg_match('@tls://@i', $host);
594 
595  $host = preg_replace('@tcp://@i', '', $host); // Remove prefix
596  $host = preg_replace('@ssl://@i', '', $host); // Remove prefix
597  $host = preg_replace('@tls://@i', '', $host); // Remove prefix
598 
599  if ($usetls && !empty($conf->global->MAIN_SMTPS_ADD_TLS_TO_HOST_FOR_HELO)) {
600  $host = 'tls://'.$host;
601  }
602 
603  $hosth = $host;
604 
605  if (!empty($conf->global->MAIL_SMTP_USE_FROM_FOR_HELO)) {
606  // If the from to is 'aaa <bbb@ccc.com>', we will keep 'ccc.com'
607  $hosth = $this->getFrom('addr');
608  $hosth = preg_replace('/^.*</', '', $hosth);
609  $hosth = preg_replace('/>.*$/', '', $hosth);
610  $hosth = preg_replace('/.*@/', '', $hosth);
611  }
612 
613  $_retVal = $this->socket_send_str('HELO '.$hosth, '250');
614  }
615 
616  // Well, did we get to the server?
617  if ($_retVal) {
618  // From this point onward most server response codes should be 250
619  // Specify who the mail is from....
620  // This has to be the raw email address, strip the "name" off
621  $resultmailfrom = $this->socket_send_str('MAIL FROM: '.$this->getFrom('addr'), '250');
622  if (!$resultmailfrom) {
623  fclose($this->socket);
624  return false;
625  }
626 
627  // 'RCPT TO:' must be given a single address, so this has to loop
628  // through the list of addresses, regardless of TO, CC or BCC
629  // and send it out "single file"
630  foreach ($this->get_RCPT_list() as $_address) {
631  /* Note:
632  * BCC email addresses must be listed in the RCPT TO command list,
633  * but the BCC header should not be printed under the DATA command.
634  * http://stackoverflow.com/questions/2750211/sending-bcc-emails-using-a-smtp-server
635  */
636 
637  /*
638  * TODO
639  * After each 'RCPT TO:' is sent, we need to make sure it was kosher,
640  * if not, the whole message will fail
641  * If any email address fails, we will need to RESET the connection,
642  * mark the last address as "bad" and start the address loop over again.
643  * If any address fails, the entire message fails.
644  */
645  $this->socket_send_str('RCPT TO: <'.$_address.'>', '250');
646  }
647 
648  // Tell the server we are ready to start sending data
649  // with any custom headers...
650  // This is the last response code we look for until the end of the message.
651  $this->socket_send_str('DATA', '354');
652 
653  // Now we are ready for the message...
654  // Ok, all the ingredients are mixed in let's cook this puppy...
655  $this->socket_send_str($this->getHeader().$this->getBodyContent()."\r\n".'.', '250');
656 
657  // Now tell the server we are done and close the socket...
658  fputs($this->socket, 'QUIT');
659  fclose($this->socket);
660  }
661  }
662 
663  return $_retVal;
664  }
