235  public function listPrinters()
236  {
237  global $conf;
238  $error = 0;
239  $line = 0;
240  $obj = array();
241  $sql = 'SELECT rowid, name, fk_type, fk_profile, parameter';
242  $sql .= ' FROM '.MAIN_DB_PREFIX.'printer_receipt';
243  $sql .= ' WHERE entity = '.$conf->entity;
244  $resql = $this->db->query($sql);
245  if ($resql) {
246  $num = $this->db->num_rows($resql);
247  while ($line < $num) {
248  $row = $this->db->fetch_array($resql);
249  switch ($row['fk_type']) {
250  case 1:
251  $row['fk_type_name'] = 'CONNECTOR_DUMMY';
252  break;
253  case 2:
254  $row['fk_type_name'] = 'CONNECTOR_FILE_PRINT';
255  break;
256  case 3:
257  $row['fk_type_name'] = 'CONNECTOR_NETWORK_PRINT';
258  break;
259  case 4:
260  $row['fk_type_name'] = 'CONNECTOR_WINDOWS_PRINT';
261  break;
262  case 5:
263  $row['fk_type_name'] = 'CONNECTOR_CUPS_PRINT';
264  break;
265  default:
266  $row['fk_type_name'] = 'CONNECTOR_UNKNOWN';
267  break;
268  }
269  switch ($row['fk_profile']) {
270  case 0:
271  $row['fk_profile_name'] = 'PROFILE_DEFAULT';
272  break;
273  case 1:
274  $row['fk_profile_name'] = 'PROFILE_SIMPLE';
275  break;
276  case 2:
277  $row['fk_profile_name'] = 'PROFILE_EPOSTEP';
278  break;
279  case 3:
280  $row['fk_profile_name'] = 'PROFILE_P822D';
281  break;
282  default:
283  $row['fk_profile_name'] = 'PROFILE_STAR';
284  break;
285  }
286  $obj[] = $row;
287  $line++;
288  }
289  } else {
290  $error++;
291  $this->errors[] = $this->db->lasterror;
292  }
293  $this->listprinters = $obj;
294  return $error;
295  }
