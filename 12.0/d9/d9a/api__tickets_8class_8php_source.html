133  private function getCommon($id = 0, $track_id = '', $ref = '')
134  {
135  if (!DolibarrApiAccess::$user->rights->ticket->read) {
136  throw new RestException(403);
137  }
138 
139  // Check parameters
140  if (!$id && !$track_id && !$ref) {
141  throw new RestException(401, 'Wrong parameters');
142  }
143 
144  $result = $this->ticket->fetch($id, $ref, $track_id);
145  if (!$result) {
146  throw new RestException(404, 'Ticket not found');
147  }
148 
149  // String for user assigned
150  if ($this->ticket->fk_user_assign > 0) {
151  $userStatic = new User($this->db);
152  $userStatic->fetch($this->ticket->fk_user_assign);
153  $this->ticket->fk_user_assign_string = $userStatic->firstname.' '.$userStatic->lastname;
154  }
155 
156  // Messages of ticket
157  $messages = array();
158  $this->ticket->loadCacheMsgsTicket();
159  if (is_array($this->ticket->cache_msgs_ticket) && count($this->ticket->cache_msgs_ticket) > 0) {
160  $num = count($this->ticket->cache_msgs_ticket);
161  $i = 0;
162  while ($i < $num) {
163  if ($this->ticket->cache_msgs_ticket[$i]['fk_user_author'] > 0) {
164  $user_action = new User($this->db);
165  $user_action->fetch($this->ticket->cache_msgs_ticket[$i]['fk_user_author']);
166  }
167 
168  // Now define messages
169  $messages[] = array(
170  'id' => $this->ticket->cache_msgs_ticket[$i]['id'],
171  'fk_user_action' => $this->ticket->cache_msgs_ticket[$i]['fk_user_author'],
172  'fk_user_action_socid' => $user_action->socid,
173  'fk_user_action_string' => dolGetFirstLastname($user_action->firstname, $user_action->lastname),
174  'message' => $this->ticket->cache_msgs_ticket[$i]['message'],
175  'datec' => $this->ticket->cache_msgs_ticket[$i]['datec'],
176  'private' => $this->ticket->cache_msgs_ticket[$i]['private']
177  );
178  $i++;
179  }
180  $this->ticket->messages = $messages;
181  }
182 
183  // History
184  $history = array();
185  $this->ticket->loadCacheLogsTicket();
186  if (is_array($this->ticket->cache_logs_ticket) && count($this->ticket->cache_logs_ticket) > 0) {
187  $num = count($this->ticket->cache_logs_ticket);
188  $i = 0;
189  while ($i < $num) {
190  if ($this->ticket->cache_logs_ticket[$i]['fk_user_create'] > 0) {
191  $user_action = new User($this->db);
192  $user_action->fetch($this->ticket->cache_logs_ticket[$i]['fk_user_create']);
193  }
194 
195  // Now define messages
196  $history[] = array(
197  'id' => $this->ticket->cache_logs_ticket[$i]['id'],
198  'fk_user_author' => $this->ticket->cache_msgs_ticket[$i]['fk_user_author'],
199  'fk_user_action' => $this->ticket->cache_logs_ticket[$i]['fk_user_create'],
200  'fk_user_action_string' => dolGetFirstLastname($user_action->firstname, $user_action->lastname),
201  'message' => $this->ticket->cache_logs_ticket[$i]['message'],
202  'datec' => $this->ticket->cache_logs_ticket[$i]['datec'],
203  );
204  $i++;
205  }
206  $this->ticket->history = $history;
207  }
208 
209 
210  if (!DolibarrApi::_checkAccessToResource('ticket', $this->ticket->id)) {
211  throw new RestException(401, 'Access not allowed for login '.DolibarrApiAccess::$user->login);
212  }
213  return $this->_cleanObjectDatas($this->ticket);

231  public function index($socid = 0, $sortfield = "t.rowid", $sortorder = "ASC", $limit = 100, $page = 0, $sqlfilters = '')
232  {
233  global $db, $conf;
234 
235  $obj_ret = array();
236 
237  if (!$socid && DolibarrApiAccess::$user->socid) {
238  $socid = DolibarrApiAccess::$user->socid;
239  }
240 
241  // If the internal user must only see his customers, force searching by him
242  if (!DolibarrApiAccess::$user->rights->societe->client->voir && !$socid) {
243  $search_sale = DolibarrApiAccess::$user->id;
244  }
245 
246  $sql = "SELECT t.rowid";
247  if ((!DolibarrApiAccess::$user->rights->societe->client->voir && !$socid) || $search_sale > 0) {
248  $sql .= ", sc.fk_soc, sc.fk_user"; // We need these fields in order to filter by sale (including the case where the user can only see his prospects)
249  }
250  $sql .= " FROM ".MAIN_DB_PREFIX."ticket as t";
251 
252  if ((!DolibarrApiAccess::$user->rights->societe->client->voir && !$socid) || $search_sale > 0) {
253  $sql .= ", ".MAIN_DB_PREFIX."societe_commerciaux as sc"; // We need this table joined to the select in order to filter by sale
254  }
255 
256  $sql .= ' WHERE t.entity IN ('.getEntity('ticket', 1).')';
257  if ((!DolibarrApiAccess::$user->rights->societe->client->voir && !$socid) || $search_sale > 0) {
258  $sql .= " AND t.fk_soc = sc.fk_soc";
259  }
260  if ($socid > 0) {
261  $sql .= " AND t.fk_soc = ".$socid;
262  }
263  if ($search_sale > 0) {
264  $sql .= " AND t.rowid = sc.fk_soc"; // Join for the needed table to filter by sale
265  }
266 
267  // Insert sale filter
268  if ($search_sale > 0) {
269  $sql .= " AND sc.fk_user = ".$search_sale;
270  }
271  // Add sql filters
272  if ($sqlfilters) {
273  if (!DolibarrApi::_checkFilters($sqlfilters)) {
274  throw new RestException(503, 'Error when validating parameter sqlfilters '.$sqlfilters);
275  }
276  $regexstring = '\(([^:\'\(\)]+:[^:\'\(\)]+:[^:\(\)]+)\)';
277  $sql .= " AND (".preg_replace_callback('/'.$regexstring.'/', 'DolibarrApi::_forge_criteria_callback', $sqlfilters).")";
278  }
279 
280  $sql .= $db->order($sortfield, $sortorder);
281 
282  if ($limit) {
283  if ($page < 0) {
284  $page = 0;
285  }
286  $offset = $limit * $page;
287 
288  $sql .= $this->db->plimit($limit, $offset);
289  }
290 
291  $result = $db->query($sql);
292  if ($result) {
293  $num = $db->num_rows($result);
294  $i = 0;
295  while ($i < $num) {
296  $obj = $db->fetch_object($result);
297  $ticket_static = new Ticket($db);
298  if ($ticket_static->fetch($obj->rowid)) {
299  if ($ticket_static->fk_user_assign > 0) {
300  $userStatic = new User($this->db);
301  $userStatic->fetch($ticket_static->fk_user_assign);
302  $ticket_static->fk_user_assign_string = $userStatic->firstname.' '.$userStatic->lastname;
303  }
304  $obj_ret[] = $this->_cleanObjectDatas($ticket_static);
305  }
306  $i++;
307  }
308  } else {
309  throw new RestException(503, 'Error when retrieve ticket list');
310  }
311  if (!count($obj_ret)) {
312  throw new RestException(404, 'No ticket found');
313  }
314  return $obj_ret;
315  }
