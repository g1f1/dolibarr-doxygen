321  public function updateProperties(Product $parent, User $user)
322  {
323  global $conf;
324 
325  $this->db->begin();
326 
327  $child = new Product($this->db);
328  $child->fetch($this->fk_product_child);
329 
330  $child->price_autogen = $parent->price_autogen;
331  $child->weight = $parent->weight;
332  if ($this->variation_weight) { // If we must add a delta on weight
333  $child->weight = ($child->weight ? $child->weight : 0) + $this->variation_weight;
334  }
335  $child->weight_units = $parent->weight_units;
336 
337  // Don't update the child label if the user has already modified it.
338  if ($child->label == $parent->label) {
339  // This will trigger only at variant creation time
340  $varlabel = $this->getCombinationLabel($this->fk_product_child);
341  $child->label = $parent->label.$varlabel;;
342  }
343 
344  if ($child->update($child->id, $user) > 0) {
345  $new_vat = $parent->tva_tx;
346  $new_npr = $parent->tva_npr;
347 
348  // MultiPrix
349  if (!empty($conf->global->PRODUIT_MULTIPRICES)) {
350  for ($i = 1; $i <= $conf->global->PRODUIT_MULTIPRICES_LIMIT; $i++)
351  {
352  if ($parent->multiprices[$i] != '') {
353  $new_type = $parent->multiprices_base_type[$i];
354  $new_min_price = $parent->multiprices_min[$i];
355  if ($parent->prices_by_qty_list[$i]) {
356  $new_psq = 1;
357  } else {
358  $new_psq = 0;
359  }
360 
361  if ($new_type == 'TTC') {
362  $new_price = $parent->multiprices_ttc[$i];
363  } else {
364  $new_price = $parent->multiprices[$i];
365  }
366 
367  if ($this->variation_price_percentage) {
368  if ($new_price != 0) {
369  $new_price *= 1 + ($this->variation_price / 100);
370  }
371  } else {
372  $new_price += $this->variation_price;
373  }
374 
375  $child->updatePrice($new_price, $new_type, $user, $new_vat, $new_min_price, $i, $new_npr, $new_psq);
376  }
377  }
378  } else {
379  $new_type = $parent->price_base_type;
380  $new_min_price = $parent->price_min;
381  $new_psq = $parent->price_by_qty;
382 
383  if ($new_type == 'TTC') {
384  $new_price = $parent->price_ttc;
385  } else {
386  $new_price = $parent->price;
387  }
388 
389  if ($this->variation_price_percentage) {
390  if ($new_price != 0) {
391  $new_price *= 1 + ($this->variation_price / 100);
392  }
393  } else {
394  $new_price += $this->variation_price;
395  }
396 
397  $child->updatePrice($new_price, $new_type, $user, $new_vat, $new_min_price, 1, $new_npr, $new_psq);
398  }
399 
400  $this->db->commit();
401 
402  return 1;
403  }
404 
405  $this->db->rollback();
406  return -1;

615  //print 'Create prod '.$newproduct->ref.'<br>'."\n";
