42 
43  if (empty($conf->global->MAIN_DISABLE_CONTACTS_TAB))

50  $h++;
51  }
52 
53  // Show more tabs from modules
54  // Entries must be declared in modules descriptor with line
55  // $this->tabs = array('entity:+tabname:Title:@mymodule:/mymodule/mypage.php?id=__ID__'); to add new tab

57  complete_head_from_modules($conf, $langs, $object, $head, $h, 'contract');
58 
59  if (empty($conf->global->MAIN_DISABLE_NOTES_TAB))
60  {
61  $nbNote = 0;
62  if (!empty($object->note_private)) $nbNote++;
63  if (!empty($object->note_public)) $nbNote++;
64  $head[$h][0] = DOL_URL_ROOT.'/contrat/note.php?id='.$object->id;
65  $head[$h][1] = $langs->trans("Notes");
66  if ($nbNote > 0) $head[$h][1] .= '<span class="badge marginleftonlyshort">'.$nbNote.'</span>';
67  $head[$h][2] = 'note';
68  $h++;
69  }
70 
71  require_once DOL_DOCUMENT_ROOT.'/core/lib/files.lib.php';
72  require_once DOL_DOCUMENT_ROOT.'/core/class/link.class.php';
73  $upload_dir = $conf->contrat->dir_output."/".dol_sanitizeFileName($object->ref);
74  $nbFiles = count(dol_dir_list($upload_dir, 'files', 0, '', '(\.meta|_preview.*\.png)$'));
75  $nbLinks = Link::count($db, $object->element, $object->id);
76  $head[$h][0] = DOL_URL_ROOT.'/contrat/document.php?id='.$object->id;
77  $head[$h][1] = $langs->trans("Documents");

79  $head[$h][2] = 'documents';
80  $h++;
81 
82  $head[$h][0] = DOL_URL_ROOT.'/contrat/agenda.php?id='.$object->id;
83  $head[$h][1] .= $langs->trans("Events");
84  if (!empty($conf->agenda->enabled) && (!empty($user->rights->agenda->myactions->read) || !empty($user->rights->agenda->allactions->read)))
85  {
86  $head[$h][1] .= '/';

89  $head[$h][2] = 'agenda';
90  $h++;
91 
92  complete_head_from_modules($conf, $langs, $object, $head, $h, 'contract', 'remove');
