164  public function connect_bind()
165  {
166  // phpcs:enable
167  global $langs, $conf;
168 
169  $connected = 0;
170  $this->bind = 0;
171 
172  // Check parameters
173  if (count($this->server) == 0 || empty($this->server[0]))
174  {
175  $this->error = 'LDAP setup (file conf.php) is not complete';
176  dol_syslog(get_class($this)."::connect_bind ".$this->error, LOG_WARNING);
177  return -1;
178  }
179 
180  if (!function_exists("ldap_connect"))
181  {
182  $this->error = 'LDAPFunctionsNotAvailableOnPHP';
183  dol_syslog(get_class($this)."::connect_bind ".$this->error, LOG_WARNING);
184  $return = -1;
185  }
186 
187  if (empty($this->error))
188  {
189  // Loop on each ldap server
190  foreach ($this->server as $key => $host)
191  {
192  if ($connected) break;
193  if (empty($host)) continue;
194 
195  if ($this->serverPing($host, $this->serverPort) === true) {
196  $this->connection = ldap_connect($host, $this->serverPort);
197  }
198  else continue;
199 
200  if (is_resource($this->connection))
201  {
202  // Begin TLS if requested by the configuration
203  if (!empty($conf->global->LDAP_SERVER_USE_TLS))
204  {
205  if (!ldap_start_tls($this->connection))
206  {
207  dol_syslog(get_class($this)."::connect_bind failed to start tls", LOG_WARNING);
208  $connected = 0;
209  $this->close();
210  }
211  }
212 
213  // Execute the ldap_set_option here (after connect and before bind)
214  $this->setVersion();
215  ldap_set_option($this->connection, LDAP_OPT_SIZELIMIT, 0); // no limit here. should return true.
216 
217 
218  if ($this->serverType == "activedirectory")
219  {
220  $result = $this->setReferrals();
221  dol_syslog(get_class($this)."::connect_bind try bindauth for activedirectory on ".$host." user=".$this->searchUser." password=".preg_replace('/./', '*', $this->searchPassword), LOG_DEBUG);
222  $this->result = $this->bindauth($this->searchUser, $this->searchPassword);
223  if ($this->result)
224  {
225  $this->bind = $this->result;
226  $connected = 2;
227  break;
228  }
229  else
230  {
231  $this->error = ldap_errno($this->connection).' '.ldap_error($this->connection);
232  }
233  }
234  else
235  {
236  // Try in auth mode
237  if ($this->searchUser && $this->searchPassword)
238  {
239  dol_syslog(get_class($this)."::connect_bind try bindauth on ".$host." user=".$this->searchUser." password=".preg_replace('/./', '*', $this->searchPassword), LOG_DEBUG);
240  $this->result = $this->bindauth($this->searchUser, $this->searchPassword);
241  if ($this->result)
242  {
243  $this->bind = $this->result;
244  $connected = 2;
245  break;
246  }
247  else
248  {
249  $this->error = ldap_errno($this->connection).' '.ldap_error($this->connection);
250  }
251  }
252  // Try in anonymous
253  if (!$this->bind)
254  {
255  dol_syslog(get_class($this)."::connect_bind try bind on ".$host, LOG_DEBUG);
256  $result = $this->bind();
257  if ($result)
258  {
259  $this->bind = $this->result;
260  $connected = 1;
261  break;
262  }
263  else
264  {
265  $this->error = ldap_errno($this->connection).' '.ldap_error($this->connection);
266  }
267  }
268  }
269  }
270 
271  if (!$connected) $this->close();
272  }
273  }
274 
275  if ($connected)
276  {
277  $return = $connected;
278  dol_syslog(get_class($this)."::connect_bind return=".$return, LOG_DEBUG);
279  }
280  else
281  {
282  $this->error = 'Failed to connect to LDAP'.($this->error ? ': '.$this->error : '');
283  $return = -1;
284  dol_syslog(get_class($this)."::connect_bind return=".$return.' - '.$this->error, LOG_WARNING);
285  }
286  return $return;
287  }

576  public function update($dn, $info, $user, $olddn, $newrdn = false, $newparent = false)
577  {
578  dol_syslog(get_class($this)."::update dn=".$dn." olddn=".$olddn);
579 
580  // Check parameters
581  if (!$this->connection)
582  {
583  $this->error = "NotConnected";
584  return -2;
585  }
586  if (!$this->bind)
587  {
588  $this->error = "NotConnected";
589  return -3;
590  }
591 
592  if (!$olddn || $olddn != $dn)
593  {
594  if (! empty($olddn) && ! empty($newrdn) && ! empty($newparent) && $this->ldapProtocolVersion === '3')
595  {
596  // This function currently only works with LDAPv3
597  $result = $this->rename($olddn, $newrdn, $newparent, $user, true);
598  }
599  else
600  {
601  // If change we make is rename the key of LDAP record, we create new one and if ok, we delete old one.
602  $result = $this->add($dn, $info, $user);
603  if ($result > 0 && $olddn && $olddn != $dn) $result = $this->delete($olddn); // If add fails, we do not try to delete old one
604  }
605  }
606  else
607  {
608  //$result = $this->delete($olddn);
609  $result = $this->add($dn, $info, $user); // If record has been deleted from LDAP, we recreate it. We ignore error if it already exists.
610  $result = $this->modify($dn, $info, $user); // We use add/modify instead of delete/add when olddn is received
611  }
612  if ($result <= 0)
613  {
614  $this->error = ldap_error($this->connection).' (Code '.ldap_errno($this->connection).") ".$this->error;
615  dol_syslog(get_class($this)."::update ".$this->error, LOG_ERR);
616  //print_r($info);
617  return -1;
618  }
619  else
620  {
621  dol_syslog(get_class($this)."::update done successfully");
622  return 1;
623  }
624  }

744  public function serverPing($host, $port = 389, $timeout = 1)
745  {
746  // Replace ldaps:// by ssl://
747  if (preg_match('/^ldaps:\/\/([^\/]+)\/?$/', $host, $regs)) {
748  $host = 'ssl://'.$regs[1];
749  }
750  // Remove ldap://
751  if (preg_match('/^ldap:\/\/([^\/]+)\/?$/', $host, $regs)) {
752  $host = $regs[1];
753  }
754  $op = @fsockopen($host, $port, $errno, $errstr, $timeout);
755  if (!$op) return false; //DC is N/A
756  else {
757  fclose($op); //explicitly close open socket connection
758  return true; //DC is up & running, we can safely connect with ldap_connect
