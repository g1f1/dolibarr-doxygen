326  public function query($query, $usesavepoint = 0, $type = 'auto')
327  {
328  $query = trim($query);
329 
330  if (preg_match('/^--/', $query)) return true;
331 
332  // Conversion syntaxe MySql vers MSDE.
333  $query = str_ireplace("now()", "getdate()", $query);
334  // Erreur SQL: cannot update timestamp field
335  $query = str_ireplace(", tms = tms", "", $query);
336 
337  $query = preg_replace("/([. ,\t(])(percent|file|public)([. ,=\t)])/", "$1[$2]$3", $query);
338 
339  $original_query = '';
340 
341  if ($type == "auto" || $type = 'dml')
342  {
343  $query = preg_replace('/AUTO_INCREMENT/i', 'IDENTITY', $query);
344  $query = preg_replace('/double/i', 'float', $query);
345  $query = preg_replace('/float\((.*)\)/', 'numeric($1)', $query);
346  $query = preg_replace('/([ \t])unsigned|IF NOT EXISTS[ \t]/i', '$1', $query);
347  $query = preg_replace('/([ \t])(MEDIUM|TINY|LONG){0,1}TEXT([ \t,])/i', "$1VARCHAR(MAX)$3", $query);
348 
349  $matches = array();
350  if (preg_match('/ALTER TABLE\h+(\w+?)\h+ADD\h+(?:(UNIQUE)|INDEX)\h+(?:INDEX)?\h*(\w+?)\h*\((.+)\)/is', $query, $matches))
351  {
352  $original_query = $query;
353  $query = "CREATE ".trim($matches[2])." INDEX [".trim($matches[3])."] ON [".trim($matches[1])."] (".trim($matches[4]).")";
354  if ($matches[2]) {
355  //check if columun is nullable cause Sql server only allow 1 null value if unique index.
356  $fields = explode(",", trim($matches[4]));
357  $fields_clear = array_map('trim', $fields);
358  $infos = $this->GetFieldInformation(trim($matches[1]), $fields_clear);
359  $query_comp = array();
360  if (is_array($infos)) {
361  foreach ($infos as $fld) {
362  if ($fld->IS_NULLABLE == 'YES') {
363  $query_comp[] = $fld->COLUMN_NAME." IS NOT NULL";
364  }
365  }
366  }
367  if (!empty($query_comp))
368  $query .= " WHERE ".implode(" AND ", $query_comp);
369  }
370  }
371  else
372  {
373  if (preg_match('/ALTER TABLE\h+(\w+?)\h+ADD\h+PRIMARY\h+KEY\h+(\w+?)\h*\((.+)\)/is', $query, $matches))
374  {
375  $original_query = $query;
376  $query = "ALTER TABLE [".$matches[1]."] ADD CONSTRAINT [".$matches[2]."] PRIMARY KEY CLUSTERED (".$matches[3].")";
377  }
378  }
379  }
380 
381  if ($type == "auto" || $type = 'ddl')
382  {
383  $itemfound = stripos($query, " limit ");
384  if ($itemfound !== false) {
385  // Extraire le nombre limite
386  $number = stristr($query, " limit ");
387  $number = substr($number, 7);
388  // Inserer l'instruction TOP et le nombre limite
389  $query = str_ireplace("select ", "select top ".$number." ", $query);
390  // Supprimer l'instruction MySql
391  $query = str_ireplace(" limit ".$number, "", $query);
392  }
393 
394  $itemfound = stripos($query, " week(");
395  if ($itemfound !== false) {
396  // Recreer une requete sans instruction Mysql
397  $positionMySql = stripos($query, " week(");
398  $newquery = substr($query, 0, $positionMySql);
399 
400  // Recuperer la date passee en parametre
401  $extractvalue = stristr($query, " week(");
402  $extractvalue = substr($extractvalue, 6);
403  $positionMySql = stripos($extractvalue, ")");
404  // Conserver la fin de la requete
405  $endofquery = substr($extractvalue, $positionMySql);
406  $extractvalue = substr($extractvalue, 0, $positionMySql);
407 
408  // Remplacer l'instruction MySql en Sql Server
409  // Inserer la date en parametre et le reste de la requete
410  $query = $newquery." DATEPART(week, ".$extractvalue.$endofquery;
411  }
412  if (preg_match('/^insert\h+(?:INTO)?\h*(\w+?)\h*\(.*\b(?:row)?id\b.*\)\h+VALUES/i', $query, $matches))
413  {
414  //var_dump($query);
415  //var_dump($matches);
416  //if (stripos($query,'llx_c_departements') !== false) var_dump($query);
417  $sql = 'SET IDENTITY_INSERT ['.trim($matches[1]).'] ON;';
418  @mssql_query($sql, $this->db);
419  $post_query = 'SET IDENTITY_INSERT ['.trim($matches[1]).'] OFF;';
420  }
421  }
422  //print "<!--".$query."-->";
423 
424  if (!in_array($query, array('BEGIN', 'COMMIT', 'ROLLBACK')))
425  {
426  $SYSLOG_SQL_LIMIT = 10000; // limit log to 10kb per line to limit DOS attacks
427  dol_syslog('sql='.substr($query, 0, $SYSLOG_SQL_LIMIT), LOG_DEBUG);
428  }
429 
430  if (!$this->database_name)
431  {
432  // Ordre SQL ne necessitant pas de connexion a une base (exemple: CREATE DATABASE)
433  $ret = mssql_query($query, $this->db);
434  }
435  else
436  {
437  $ret = mssql_query($query, $this->db);
438  }
439 
440  if (!empty($post_query))
441  {
442  @mssql_query($post_query, $this->db);
443  }
444 
445  if (!preg_match("/^COMMIT/i", $query) && !preg_match("/^ROLLBACK/i", $query))
446  {
447  // Si requete utilisateur, on la sauvegarde ainsi que son resultset
448  if (!$ret)
449  {
450  $result = mssql_query("SELECT @@ERROR as code", $this->db);
451  $row = mssql_fetch_array($result);
452 
453  $this->lastqueryerror = $query;
454  $this->lasterror = $this->error();
455  $this->lasterrno = $row["code"];
456 
457  dol_syslog(get_class($this)."::query SQL Error query: ".$query, LOG_ERR);
458  if ($original_query) dol_syslog(get_class($this)."::query SQL Original query: ".$original_query, LOG_ERR);
459  dol_syslog(get_class($this)."::query SQL Error message: ".$this->lasterror." (".$this->lasterrno.")", LOG_ERR);
460  }
461  $this->lastquery = $query;
462  $this->_results = $ret;
463  }
464 
465  return $ret;
466  }
