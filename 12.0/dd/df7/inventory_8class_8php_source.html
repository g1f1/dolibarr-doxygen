254  // Reset some properties

256  unset($object->fk_user_creat);
257  unset($object->import_key);
258 
259  // Clear fields
260  $object->ref = "copy_of_".$object->ref;
261  $object->title = $langs->trans("CopyOf")." ".$object->title;
262  // ...
263 
264  // Create clone
265  $object->context['createfromclone'] = 'createfromclone';
266  $result = $object->createCommon($user);
267  if ($result < 0) {
268  $error++;
269  $this->error = $object->error;
270  $this->errors = $object->errors;
271  }
272 
273  unset($object->context['createfromclone']);
274 
275  // End
276  if (!$error) {
277  $this->db->commit();
278  return $object;
279  } else {
280  $this->db->rollback();
281  return -1;
282  }

428  public function info($id)
429  {
430  $sql = 'SELECT rowid, date_creation as datec, tms as datem,';
431  $sql .= ' fk_user_creat, fk_user_modif';
432  $sql .= ' FROM '.MAIN_DB_PREFIX.$this->table_element.' as t';
433  $sql .= ' WHERE t.rowid = '.$id;
434  $result = $this->db->query($sql);
435  if ($result)
436  {
437  if ($this->db->num_rows($result))
438  {
439  $obj = $this->db->fetch_object($result);
440  $this->id = $obj->rowid;
441  if ($obj->fk_user_author)
442  {
443  $cuser = new User($this->db);
444  $cuser->fetch($obj->fk_user_author);
445  $this->user_creation = $cuser;
446  }
447 
448  if ($obj->fk_user_valid)
449  {
450  $vuser = new User($this->db);
451  $vuser->fetch($obj->fk_user_valid);
452  $this->user_validation = $vuser;
453  }
454 
455  if ($obj->fk_user_cloture)
456  {
457  $cluser = new User($this->db);
458  $cluser->fetch($obj->fk_user_cloture);
459  $this->user_cloture = $cluser;
460  }
461 
462  $this->date_creation = $this->db->jdate($obj->datec);
463  $this->date_modification = $this->db->jdate($obj->datem);
464  $this->date_validation = $this->db->jdate($obj->datev);
465  }
466 
467  $this->db->free($result);
468  }
469  else

471  dol_print_error($this->db);
472  }
