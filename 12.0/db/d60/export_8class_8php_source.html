226  public function build_sql($indice, $array_selected, $array_filterValue)
227  {
228  // phpcs:enable
229  // Build the sql request
230  $sql = $this->array_export_sql_start[$indice];
231  $i = 0;
232 
233  //print_r($array_selected);
234  foreach ($this->array_export_fields[$indice] as $key => $value)
235  {
236  if (!array_key_exists($key, $array_selected)) continue; // Field not selected
237  if (preg_match('/^none\./', $key)) continue; // A field that must not appears into SQL
238  if ($i > 0) $sql .= ', ';
239  else $i++;
240 
241  if (strpos($key, ' as ') === false) {
242  $newfield = $key.' as '.str_replace(array('.', '-', '(', ')'), '_', $key);
243  } else {
244  $newfield = $key;
245  }
246 
247  $sql .= $newfield;
248  }
249  $sql .= $this->array_export_sql_end[$indice];
250 
251  // Add the WHERE part. Filtering into sql if a filtering array is provided
252  if (is_array($array_filterValue) && !empty($array_filterValue))
253  {
254  $sqlWhere = '';
255  // Loop on each condition to add
256  foreach ($array_filterValue as $key => $value)
257  {
258  if (preg_match('/GROUP_CONCAT/i', $key)) continue;
259  if ($value != '') $sqlWhere .= " and ".$this->build_filterQuery($this->array_export_TypeFields[$indice][$key], $key, $array_filterValue[$key]);
260  }
261  $sql .= $sqlWhere;
262  }
263 
264  // Add the order
265  $sql .= $this->array_export_sql_order[$indice];
266 
267  // Add the HAVING part.
268  if (is_array($array_filterValue) && !empty($array_filterValue))
269  {
270  // Loop on each condition to add
271  foreach ($array_filterValue as $key => $value)
272  {
273  if (preg_match('/GROUP_CONCAT/i', $key) and $value != '') $sql .= " HAVING ".$this->build_filterQuery($this->array_export_TypeFields[$indice][$key], $key, $array_filterValue[$key]);
274  }
275  }
276 
277  return $sql;
278  }

289  public function build_filterQuery($TypeField, $NameField, $ValueField)
290  {
291  // phpcs:enable
292  //print $TypeField." ".$NameField." ".$ValueField;
293  $InfoFieldList = explode(":", $TypeField);
294  // build the input field on depend of the type of file
295  switch ($InfoFieldList[0]) {
296  case 'Text':
297  if (!(strpos($ValueField, '%') === false))
298  $szFilterQuery .= " ".$NameField." LIKE '".$ValueField."'";
299  else
300  $szFilterQuery .= " ".$NameField." = '".$ValueField."'";
301  break;
302  case 'Date':
303  if (strpos($ValueField, "+") > 0)
304  {
305  // mode plage
306  $ValueArray = explode("+", $ValueField);
307  $szFilterQuery = "(".$this->conditionDate($NameField, trim($ValueArray[0]), ">=");
308  $szFilterQuery .= " AND ".$this->conditionDate($NameField, trim($ValueArray[1]), "<=").")";
309  }
310  else
311  {
312  if (is_numeric(substr($ValueField, 0, 1)))
313  $szFilterQuery = $this->conditionDate($NameField, trim($ValueField), "=");
314  else
315  $szFilterQuery = $this->conditionDate($NameField, trim(substr($ValueField, 1)), substr($ValueField, 0, 1));
316  }
317  break;
318  case 'Duree':
319  break;
320  case 'Numeric':
321  // si le signe -
322  if (strpos($ValueField, "+") > 0)
323  {
324  // mode plage
325  $ValueArray = explode("+", $ValueField);
326  $szFilterQuery = "(".$NameField.">=".$ValueArray[0];
327  $szFilterQuery .= " AND ".$NameField."<=".$ValueArray[1].")";
328  }
329  else
330  {
331  if (is_numeric(substr($ValueField, 0, 1)))
332  $szFilterQuery = " ".$NameField."=".$ValueField;
333  else
334  $szFilterQuery = " ".$NameField.substr($ValueField, 0, 1).substr($ValueField, 1);
335  }
336  break;
337  case 'Boolean':
338  $szFilterQuery = " ".$NameField."=".(is_numeric($ValueField) ? $ValueField : ($ValueField == 'yes' ? 1 : 0));
339  break;
340  case 'Status':
341  case 'List':
342  if (is_numeric($ValueField))
343  $szFilterQuery = " ".$NameField."=".$ValueField;
344  else {
345  if (!(strpos($ValueField, '%') === false))
346  $szFilterQuery = " ".$NameField." LIKE '".$ValueField."'";
347  else
348  $szFilterQuery = " ".$NameField." = '".$ValueField."'";
349  }
350  break;
351  default:
352  dol_syslog("Error we try to forge an sql export request with a condition on a field with type '".$InfoFieldList[0]."' (defined into module descriptor) but this type is unknown/not supported. It looks like a bug into module descriptor.", LOG_ERR);
353  }
354 
355  return $szFilterQuery;
356  }
