165  public function index($sortfield = "t.ref", $sortorder = 'ASC', $limit = 100, $page = 0, $mode = 0, $category = 0, $sqlfilters = '')
166  {
167  global $db, $conf;
168 
169  $obj_ret = array();
170 
171  $socid = DolibarrApiAccess::$user->socid ? DolibarrApiAccess::$user->socid : '';
172 
173  $sql = "SELECT t.rowid, t.ref, t.ref_ext";
174  $sql .= " FROM ".MAIN_DB_PREFIX."product as t";
175  if ($category > 0) {
176  $sql .= ", ".MAIN_DB_PREFIX."categorie_product as c";
177  }
178  $sql .= ' WHERE t.entity IN ('.getEntity('product').')';
179  // Select products of given category
180  if ($category > 0) {
181  $sql .= " AND c.fk_categorie = ".$db->escape($category);
182  $sql .= " AND c.fk_product = t.rowid ";
183  }
184  if ($mode == 1) {
185  // Show only products
186  $sql .= " AND t.fk_product_type = 0";
187  } elseif ($mode == 2) {
188  // Show only services
189  $sql .= " AND t.fk_product_type = 1";
190  }
191  // Add sql filters
192  if ($sqlfilters) {
193  if (!DolibarrApi::_checkFilters($sqlfilters)) {
194  throw new RestException(503, 'Error when validating parameter sqlfilters '.$sqlfilters);
195  }
196  $regexstring = '\(([^:\'\(\)]+:[^:\'\(\)]+:[^:\(\)]+)\)';
197  $sql .= " AND (".preg_replace_callback('/'.$regexstring.'/', 'DolibarrApi::_forge_criteria_callback', $sqlfilters).")";
198  }
199 
200  $sql .= $db->order($sortfield, $sortorder);
201  if ($limit) {
202  if ($page < 0) {
203  $page = 0;
204  }
205  $offset = $limit * $page;
206 
207  $sql .= $db->plimit($limit + 1, $offset);
208  }
209 
210  $result = $db->query($sql);
211  if ($result) {
212  $num = $db->num_rows($result);
213  $min = min($num, ($limit <= 0 ? $num : $limit));
214  $i = 0;
215  while ($i < $min)
216  {
217  $obj = $db->fetch_object($result);
218  $product_static = new Product($db);
219  if ($product_static->fetch($obj->rowid)) {
220  $obj_ret[] = $this->_cleanObjectDatas($product_static);
221  }
222  $i++;
223  }
224  }
225  else {
226  throw new RestException(503, 'Error when retrieve product list : '.$db->lasterror());
227  }
228  if (!count($obj_ret)) {
229  throw new RestException(404, 'No product found');
230  }
231  return $obj_ret;
232  }

269  public function put($id, $request_data = null)
270  {
271  global $conf;
272 
273  if (!DolibarrApiAccess::$user->rights->produit->creer) {
274  throw new RestException(401);
275  }
276 
277  $result = $this->product->fetch($id);
278  if (!$result) {
279  throw new RestException(404, 'Product not found');
280  }
281 
282  if (!DolibarrApi::_checkAccessToResource('product', $this->product->id)) {
283  throw new RestException(401, 'Access not allowed for login '.DolibarrApiAccess::$user->login);
284  }
285 
286  $oldproduct = dol_clone($this->product, 0);
287 
288  foreach ($request_data as $field => $value) {
289  if ($field == 'id') { continue;
290  }
291  $this->product->$field = $value;
292  }
293 
294  $updatetype = false;
295  if ($this->product->type != $oldproduct->type && ($this->product->isProduct() || $this->product->isService())) {
296  $updatetype = true;
297  }
298 
299  $result = $this->product->update($id, DolibarrApiAccess::$user, 1, 'update', $updatetype);
300 
301  // If price mode is 1 price per product
302  if ($result > 0 && !empty($conf->global->PRODUCT_PRICE_UNIQ)) {
303  // We update price only if it was changed
304  $pricemodified = false;
305  if ($this->product->price_base_type != $oldproduct->price_base_type) { $pricemodified = true;
306  } else
307  {
308  if ($this->product->tva_tx != $oldproduct->tva_tx) { $pricemodified = true;
309  }
310  if ($this->product->tva_npr != $oldproduct->tva_npr) { $pricemodified = true;
311  }
312  if ($this->product->default_vat_code != $oldproduct->default_vat_code) { $pricemodified = true;
313  }
314 
315  if ($this->product->price_base_type == 'TTC') {
316  if ($this->product->price_ttc != $oldproduct->price_ttc) { $pricemodified = true;
317  }
318  if ($this->product->price_min_ttc != $oldproduct->price_min_ttc) { $pricemodified = true;
319  }
320  }
321  else
322  {
323  if ($this->product->price != $oldproduct->price) { $pricemodified = true;
324  }
325  if ($this->product->price_min != $oldproduct->price_min) { $pricemodified = true;
326  }
327  }
328  }
329 
330  if ($pricemodified) {
331  $newvat = $this->product->tva_tx;
332  $newnpr = $this->product->tva_npr;
333  $newvatsrccode = $this->product->default_vat_code;
334 
335  $newprice = $this->product->price;
336  $newpricemin = $this->product->price_min;
337  if ($this->product->price_base_type == 'TTC') {
338  $newprice = $this->product->price_ttc;
339  $newpricemin = $this->product->price_min_ttc;
340  }
341 
342  $result = $this->product->updatePrice($newprice, $this->product->price_base_type, DolibarrApiAccess::$user, $newvat, $newpricemin, 0, $newnpr, 0, 0, array(), $newvatsrccode);
343  }
344  }
345 
346  if ($result <= 0) {
347  throw new RestException(500, "Error updating product", array_merge(array($this->product->error), $this->product->errors));
348  }
349 
350  return $this->get($id);
351  }
