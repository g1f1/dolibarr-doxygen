57  public function selectWebsite($selected = '', $htmlname = 'exportmodelid', $useempty = 0)
58  {
59  $out = '';
60 
61  $sql = "SELECT rowid, ref";
62  $sql .= " FROM ".MAIN_DB_PREFIX."website";
63  $sql .= " WHERE 1 = 1";
64  $sql .= " ORDER BY rowid";
65  $result = $this->db->query($sql);
66  if ($result)
67  {
68  $out .= '<select class="flat minwidth100" name="'.$htmlname.'" id="'.$htmlname.'">';
69  if ($useempty)
70  {
71  $out .= '<option value="-1">&nbsp;</option>';
72  }
73 
74  $num = $this->db->num_rows($result);
75  $i = 0;
76  while ($i < $num)
77  {
78  $obj = $this->db->fetch_object($result);
79  if ($selected == $obj->rowid)
80  {
81  $out .= '<option value="'.$obj->rowid.'" selected>';
82  }
83  else
84  {
85  $out .= '<option value="'.$obj->rowid.'">';
86  }
87  $out .= $obj->ref;
88  $out .= '</option>';
89  $i++;
90  }
91  $out .= "</select>";
92  }
93  else {
94  dol_print_error($this->db);
95  }
96 
97  return $out;
98  }

235  public function selectContainer($website, $htmlname = 'pageid', $pageid = 0, $showempty = 0, $action = '', $morecss = 'minwidth200', $excludeids = null)
236  {
237  global $langs;
238 
239  $atleastonepage = (is_array($website->lines) && count($website->lines) > 0);
240 
241  $out = '';
242  if ($atleastonepage && $action != 'editsource')
243  {
244  $out .= '<select name="'.$htmlname.'" id="'.$htmlname.'" class="maxwidth300'.($morecss ? ' '.$morecss : '').'">';
245  }
246  else
247  {
248  $out .= '<select name="pageidbis" id="pageid" class="maxwidth300'.($morecss ? ' '.$morecss : '').'" disabled="disabled">';
249  }
250 
251  if ($showempty || !$atleastonepage) $out .= '<option value="-1">&nbsp;</option>';
252 
253  if ($atleastonepage)
254  {
255  if (empty($pageid) && $action != 'createcontainer') // Page id is not defined, we try to take one
256  {
257  $firstpageid = 0; $homepageid = 0;
258  foreach ($website->lines as $key => $valpage)
259  {
260  if (empty($firstpageid)) $firstpageid = $valpage->id;
261  if ($website->fk_default_home && $key == $website->fk_default_home) $homepageid = $valpage->id;
262  }
263  $pageid = $homepageid ? $homepageid : $firstpageid; // We choose home page and if not defined yet, we take first page
264  }
265 
266  foreach ($website->lines as $key => $valpage)
267  {
268  if (is_array($excludeids) && count($excludeids) && in_array($valpage->id, $excludeids)) continue;
269 
270  $valueforoption = '<span class="opacitymedium">['.$valpage->type_container.' '.sprintf("%03d", $valpage->id).']</span> ';
271  $valueforoption .= $valpage->pageurl.' - '.$valpage->title;
272  if ($website->otherlang) { // If there is alternative lang for this web site, we show the language code
273  if ($valpage->lang) {
274  $valueforoption .= ' <span class="opacitymedium">('.$valpage->lang.')</span>';
275  }
276  }
277  if ($website->fk_default_home && $key == $website->fk_default_home) {
278  //$valueforoption .= ' <span class="opacitymedium">('.$langs->trans("HomePage").')</span>';
279  $valueforoption .= ' <span class="opacitymedium fa fa-home"></span>';
280  }
281 
282  $out .= '<option value="'.$key.'"';
283  if ($pageid > 0 && $pageid == $key) $out .= ' selected'; // To preselect a value
284  $out .= ' data-html="'.dol_escape_htmltag($valueforoption).'"';
285  $out .= '>';
286  $out .= $valueforoption;
287  $out .= '</option>';
288  }
289  }
290  $out .= '</select>';
291 
292  if ($atleastonepage && $action != 'editsource')
293  {
294  $out .= ajax_combobox($htmlname);
295  }
296  else
297  {
298  $out .= '<input type="hidden" name="'.$htmlname.'" value="'.$pageid.'">';
299  $out .= ajax_combobox($htmlname);
300  }
301  return $out;
302  }
