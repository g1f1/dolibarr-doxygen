301  public function write_record($array_selected_sorted, $objp, $outputlangs, $array_types)
302  {
303  // phpcs:enable
304  global $conf;
305 
306  // Define first row
307  $this->col = 1;
308  if (!empty($conf->global->MAIN_USE_PHP_WRITEEXCEL)) {
309  $this->col = 0;
310  }
311 
312  $reg = array();
313 
314  foreach ($array_selected_sorted as $code => $value)
315  {
316  if (strpos($code, ' as ') == 0) $alias = str_replace(array('.', '-', '(', ')'), '_', $code);
317  else $alias = substr($code, strpos($code, ' as ') + 4);
318  if (empty($alias)) dol_print_error('', 'Bad value for field with code='.$code.'. Try to redefine export.');
319  $newvalue = $objp->$alias;
320 
321  $newvalue = $this->excel_clean($newvalue);
322  $typefield = isset($array_types[$code]) ? $array_types[$code] : '';
323 
324  if (preg_match('/^Select:/i', $typefield, $reg) && $typefield = substr($typefield, 7))
325  {
326  $array = unserialize($typefield);
327  $array = $array['options'];
328  $newvalue = $array[$newvalue];
329  }
330 
331  // Traduction newvalue
332  if (preg_match('/^\((.*)\)$/i', $newvalue, $reg))
333  {
334  $newvalue = $outputlangs->transnoentities($reg[1]);
335  }
336  else
337  {
338  $newvalue = $outputlangs->convToOutputCharset($newvalue);
339  }
340 
341  if (preg_match('/^[0-9][0-9][0-9][0-9]-[0-9][0-9]-[0-9][0-9]$/i', $newvalue))
342  {
343  $newvalue = dol_stringtotime($newvalue);
344  $this->workbook->getActiveSheet()->SetCellValueByColumnAndRow($this->col, $this->row + 1, \PhpOffice\PhpSpreadsheet\Shared\Date::PHPToExcel($newvalue));
345  $coord = $this->workbook->getActiveSheet()->getCellByColumnAndRow($this->col, $this->row + 1)->getCoordinate();
346  $this->workbook->getActiveSheet()->getStyle($coord)->getNumberFormat()->setFormatCode('yyyy-mm-dd');
347  }
348  elseif (preg_match('/^[0-9][0-9][0-9][0-9]-[0-9][0-9]-[0-9][0-9] [0-9][0-9]:[0-9][0-9]:[0-9][0-9]$/i', $newvalue))
349  {
350  $newvalue = dol_stringtotime($newvalue);

352  $coord = $this->workbook->getActiveSheet()->getCellByColumnAndRow($this->col, $this->row + 1)->getCoordinate();
353  $this->workbook->getActiveSheet()->getStyle($coord)->getNumberFormat()->setFormatCode('yyyy-mm-dd h:mm:ss');
354  }
355  else
356  {
357  if ($typefield == 'Text' || $typefield == 'TextAuto')
358  {
359  //$this->workbook->getActiveSheet()->getCellByColumnAndRow($this->col, $this->row+1)->setValueExplicit($newvalue, PHPExcel_Cell_DataType::TYPE_STRING);

364  }

368  }
369  }

371  }
372  $this->row++;
373  return 0;
