434  protected function handleFileUpload($uploaded_file, $name, $size, $type, $error, $index)
435  {
436  $file = new stdClass();
437  $file->name = $this->trimFileName($name, $type, $index);
438  $file->mime = dol_mimetype($file->name, '', 2);
439  $file->size = intval($size);
440  $file->type = $type;
441  if ($this->validate($uploaded_file, $file, $error, $index) && dol_mkdir($this->options['upload_dir']) >= 0)
442  {
443  $file_path = $this->options['upload_dir'].$file->name;
444  $append_file = !$this->options['discard_aborted_uploads'] && is_file($file_path) && $file->size > filesize($file_path);
445  clearstatcache();
446  if ($uploaded_file && is_uploaded_file($uploaded_file)) {
447  // multipart/formdata uploads (POST method uploads)
448  if ($append_file)
449  {
450  file_put_contents($file_path, fopen($uploaded_file, 'r'), FILE_APPEND);
451  } else {
452  dol_move_uploaded_file($uploaded_file, $file_path, 1, 0, 0, 0, 'userfile');
453  }
454  }
455  else
456  {
457  // Non-multipart uploads (PUT method support)
458  file_put_contents($file_path, fopen('php://input', 'r'), $append_file ? FILE_APPEND : 0);
459  }
460  $file_size = filesize($file_path);
461  if ($file_size === $file->size)
462  {
463  $file->url = $this->options['upload_url'].rawurlencode($file->name);
464  foreach ($this->options['image_versions'] as $version => $options)
465  {

471  }
472  }

474  {
475  unlink($file_path);
476  $file->error = 'abort';
477  }

479  $this->setFileDeleteUrl($file);
480  }
