130  public function create($user, $notrigger = false)
131  {
132  global $langs;
133 
134  $error = 0;
135 
136  $now = dol_now();
137 
138  // Check parameters
139  if ($this->datef <= $this->dateh)
140  {
141  $this->error = $langs->trans("ErrorBadValueForDate");
142  return -1;
143  }
144  if (empty($this->datec)) $this->datec = $now;
145 
146 
147  $this->db->begin();
148 
149  $sql = "INSERT INTO ".MAIN_DB_PREFIX."subscription (fk_adherent, fk_type, datec, dateadh, datef, subscription, note)";
150 
151  require_once DOL_DOCUMENT_ROOT.'/adherents/class/adherent.class.php';
152  $member = new Adherent($this->db);
153  $result = $member->fetch($this->fk_adherent);
154 
155  if ($this->fk_type == null) { // If type not defined, we use the type of member
156  $type = $member->typeid;
157  } else {
158  $type = $this->fk_type;
159  }
160  $sql .= " VALUES (".$this->fk_adherent.", '".$type."', '".$this->db->idate($now)."',";
161  $sql .= " '".$this->db->idate($this->dateh)."',";
162  $sql .= " '".$this->db->idate($this->datef)."',";
163  $sql .= " ".$this->amount.",";
164  $sql .= " '".$this->db->escape($this->note_public ? $this->note_public : $this->note)."')";
165 
166  $resql = $this->db->query($sql);
167  if (!$resql) {
168  $error++;
169  $this->errors[] = $this->db->lasterror();
170  }
171 
172  if (!$error)
173  {
174  $this->id = $this->db->last_insert_id(MAIN_DB_PREFIX.$this->table_element);
175  $this->fk_type = $type;
176  }
177 
178  if (!$error && !$notrigger)
179  {
180  $this->context = array('member'=>$member);
181  // Call triggers
182  $result = $this->call_trigger('MEMBER_SUBSCRIPTION_CREATE', $user);
183  if ($result < 0) { $error++; }
184  // End call triggers
185  }
186 
187  // Commit or rollback
188  if ($error) {
189  $this->db->rollback();
190  return -1;
191  } else {
192  $this->db->commit();
193  return $this->id;
194  }
195  }
