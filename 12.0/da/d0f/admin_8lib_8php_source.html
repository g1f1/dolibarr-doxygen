60 function versioncompare($versionarray1, $versionarray2)
61 {
62  $ret = 0;
63  $level = 0;
64  $count1 = count($versionarray1);
65  $count2 = count($versionarray2);
66  $maxcount = max($count1, $count2);
67  while ($level < $maxcount)
68  {
69  $operande1 = isset($versionarray1[$level]) ? $versionarray1[$level] : 0;
70  $operande2 = isset($versionarray2[$level]) ? $versionarray2[$level] : 0;
71  if (preg_match('/alpha|dev/i', $operande1)) $operande1 = -5;
72  if (preg_match('/alpha|dev/i', $operande2)) $operande2 = -5;
73  if (preg_match('/beta$/i', $operande1)) $operande1 = -4;
74  if (preg_match('/beta$/i', $operande2)) $operande2 = -4;
75  if (preg_match('/beta([0-9])+/i', $operande1)) $operande1 = -3;
76  if (preg_match('/beta([0-9])+/i', $operande2)) $operande2 = -3;
77  if (preg_match('/rc$/i', $operande1)) $operande1 = -2;
78  if (preg_match('/rc$/i', $operande2)) $operande2 = -2;
79  if (preg_match('/rc([0-9])+/i', $operande1)) $operande1 = -1;
80  if (preg_match('/rc([0-9])+/i', $operande2)) $operande2 = -1;
81  $level++;
82  //print 'level '.$level.' '.$operande1.'-'.$operande2.'<br>';
83  if ($operande1 < $operande2) { $ret = -$level; break; }
84  if ($operande1 > $operande2) { $ret = $level; break; }
85  }
86  //print join('.',$versionarray1).'('.count($versionarray1).') / '.join('.',$versionarray2).'('.count($versionarray2).') => '.$ret.'<br>'."\n";
87  return $ret;
88 }

214 
215  //print $buf.'<br>';exit;
216

219  // Found new request
220  if ($buffer) $arraysql[$i] = $buffer;
221  $i++;
222  $buffer = '';
223  }
224  }
225 
226  if ($buffer) $arraysql[$i] = $buffer;
227  fclose($fp);
228  }

231  dol_syslog("Admin.lib::run_sql failed to open file ".$sqlfile, LOG_ERR);

265  $from = '__+MAX_'.$table.'__';
266  $to = '+'.$listofmaxrowid[$table];
267  $newsql = str_replace($from, $to, $newsql);

274  {
275  // Replace lines
276  // 'INSERT INTO llx_accounting_account (entity, rowid, fk_pcg_version, pcg_type, account_number, account_parent, label, active) VALUES (__ENTITY__, 1401, 'PCG99-ABREGE', 'CAPIT', '1234', 1400, '...', 1);'

288  $cursorinsert = 0;

290  foreach ($arraysql as $i => $sql)
291  {
292  if ($sql)
293  {
294  // Replace the prefix tables
295  if (MAIN_DB_PREFIX != 'llx_')
296  {
297  $sql = preg_replace('/llx_/i', MAIN_DB_PREFIX, $sql);
298  }
299 
300  if (!empty($handler)) $sql = preg_replace('/__HANDLER__/i', "'".$handler."'", $sql);

303 
304  // Ajout trace sur requete (eventuellement a commenter si beaucoup de requetes)

306  dol_syslog('Admin.lib::run_sql Request '.($i + 1), LOG_DEBUG);
307  $sqlmodified = 0;
308 
309  // Replace for encrypt data

311  {
312  $num = count($reg[0]);
313 
314  for ($j = 0; $j < $num; $j++)
315  {
316  $from = $reg[0][$j];
317  $to = $db->encrypt($reg[1][$j], 1);
318  $newsql = str_replace($from, $to, $newsql);
319  }
320  $sqlmodified++;
321  }

324  if (preg_match_all('/__DECRYPT\(\'([A-Za-z0-9_]+)\'\)__/i', $newsql, $reg))

326  $num = count($reg[0]);

329  {

352  $sqlmodified++;
353  }

363  {
364  $cursorinsert++;
365

367  $table = preg_replace('/([^a-zA-Z_]+)/i', '', $reg[1]);
368  $insertedrowid = $db->last_insert_id($table);
369  $listofinsertedrowid[$cursorinsert] = $insertedrowid;
370  dol_syslog('Admin.lib::run_sql Insert nb '.$cursorinsert.', done in table '.$table.', rowid is '.$listofinsertedrowid[$cursorinsert], LOG_DEBUG);
371  }
372  // print '<td class="right">OK</td>';
373  }
374  else
375  {
376  $errno = $db->errno();
377  if (!$silent) print '<!-- Result = '.$errno.' -->'."\n";
378

393  'DB_ERROR_22P02'
394  );
395  if ($okerror == 'none') $okerrors = array();
396 
397  // Is it an error we accept
398  if (!in_array($errno, $okerrors))
399  {
400  if (!$silent) print '<tr><td class="tdtop" colspan="2">';
401  if (!$silent) print '<div class="error">'.$langs->trans("Error")." ".$db->errno().": ".$newsql."<br>".$db->error()."</div></td>";
402  if (!$silent) print '</tr>'."\n";
403  dol_syslog('Admin.lib::run_sql Request '.($i + 1)." Error ".$db->errno()." ".$newsql."<br>".$db->error(), LOG_ERR);
404  $error++;
405  }
406  }
407 
408  if (!$silent) print '</tr>'."\n";
409  }
410  }
411 
412  if ($error == 0)
413  {
414  if (!$silent) print '<tr><td>'.$langs->trans("ProcessMigrateScript").'</td>';
415  if (!$silent) print '<td class="right">'.$langs->trans("OK").'</td></tr>'."\n";
416  $ok = 1;
417  }
418  else

420  if (!$silent) print '<tr><td>'.$langs->trans("ProcessMigrateScript").'</td>';
421  if (!$silent) print '<td class="right"><font class="error">'.$langs->trans("KO").'</font></td></tr>'."\n";
422  $ok = 0;
423  }
