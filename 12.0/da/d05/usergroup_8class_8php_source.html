556  public function getrights($moduletag = '')
557  {
558  global $conf;
559 
560  if ($moduletag && isset($this->_tab_loaded[$moduletag]) && $this->_tab_loaded[$moduletag])
561  {
562  // Rights for this module are already loaded, so we leave
563  return;
564  }
565 
566  if (!empty($this->all_permissions_are_loaded))
567  {
568  // We already loaded all rights for this group, so we leave
569  return;
570  }
571 
572  /*
573  * Recuperation des droits
574  */
575  $sql = "SELECT r.module, r.perms, r.subperms ";
576  $sql .= " FROM ".MAIN_DB_PREFIX."usergroup_rights as u, ".MAIN_DB_PREFIX."rights_def as r";
577  $sql .= " WHERE r.id = u.fk_id";
578  $sql .= " AND r.entity = ".$conf->entity;
579  $sql .= " AND u.entity = ".$conf->entity;
580  $sql .= " AND u.fk_usergroup = ".$this->id;
581  $sql .= " AND r.perms IS NOT NULL";
582  if ($moduletag) $sql .= " AND r.module = '".$this->db->escape($moduletag)."'";
583 
584  dol_syslog(get_class($this).'::getrights', LOG_DEBUG);
585  $resql = $this->db->query($sql);
586  if ($resql)
587  {
588  $num = $this->db->num_rows($resql);
589  $i = 0;
590  while ($i < $num)
591  {
592  $obj = $this->db->fetch_object($resql);
593 
594  $module = $obj->module;
595  $perms = $obj->perms;
596  $subperms = $obj->subperms;
597 
598  if ($perms)
599  {
600  if (!isset($this->rights)) $this->rights = new stdClass(); // For avoid error
601  if (!isset($this->rights->$module) || !is_object($this->rights->$module)) $this->rights->$module = new stdClass();
602  if ($subperms)
603  {
604  if (!isset($this->rights->$module->$perms) || !is_object($this->rights->$module->$perms)) $this->rights->$module->$perms = new stdClass();
605  if (empty($this->rights->$module->$perms->$subperms)) $this->nb_rights++;
606  $this->rights->$module->$perms->$subperms = 1;
607  }
608  else
609  {
610  if (empty($this->rights->$module->$perms)) $this->nb_rights++;
611  $this->rights->$module->$perms = 1;
612  }
613  }
614 
615  $i++;
616  }
617  $this->db->free($resql);
618  }
619 
620  if ($moduletag == '')
621  {
622  // Si module etait non defini, alors on a tout charge, on peut donc considerer
623  // que les droits sont en cache (car tous charges) pour cet instance de group
624  $this->all_permissions_are_loaded = 1;
625  }
626  else
627  {
628  // If module defined, we flag it as loaded into cache
629  $this->_tab_loaded[$moduletag] = 1;
630  }
631 
632  return 1;
633  }
