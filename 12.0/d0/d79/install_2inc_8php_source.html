500 function detect_dolibarr_main_document_root()
501 {
502  // If PHP is in CGI mode, SCRIPT_FILENAME is PHP's path.
503  // Since that's not what we want, we suggest $_SERVER["DOCUMENT_ROOT"]
504  if ($_SERVER["SCRIPT_FILENAME"] == 'php' || preg_match('/[\\/]php$/i', $_SERVER["SCRIPT_FILENAME"]) || preg_match('/php\.exe$/i', $_SERVER["SCRIPT_FILENAME"]))
505  {
506  $dolibarr_main_document_root = $_SERVER["DOCUMENT_ROOT"];
507 
508  if (!preg_match('/[\\/]dolibarr[\\/]htdocs$/i', $dolibarr_main_document_root)) {
509  $dolibarr_main_document_root .= "/dolibarr/htdocs";
510  }
511  } else {
512  // We assume /install to be under /htdocs, so we get the parent directory of the current directory
513  $dolibarr_main_document_root = dirname(dirname($_SERVER["SCRIPT_FILENAME"]));

548  if (!empty($_SERVER["HTTP_HOST"])) {

552  $serverport = $_SERVER["SERVER_NAME"];

554  else {
555  $serverport = 'localhost';

558  }
