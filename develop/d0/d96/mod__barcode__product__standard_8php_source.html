336  $newcodefortest = $codefortest;
337 
338  // Special case, if mask is on 12 digits instead of 13, we remove last char into code to test
339  if (in_array($typefortest, array('EAN13', 'ISBN'))) { // We remove the CRC char not included into mask
340  if (preg_match('/\{(0+)([@\+][0-9]+)?([@\+][0-9]+)?\}/i', $mask, $reg)) {
341  if (strlen($reg[1]) == 12) {
342  $newcodefortest = substr($newcodefortest, 0, 12);
343  }

345  }

349  if (is_string($result)) {
