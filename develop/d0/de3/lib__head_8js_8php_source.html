240 /*
241  * =================================================================
242  * Function:
243  * formatDate (javascript object Date(), format) Purpose: Returns a date in the
244  * output format specified. The format string can use the following tags: Field |
245  * Tags -------------+------------------------------- Year | yyyy (4 digits), yy
246  * (2 digits) Month | MM (2 digits) Day of Month | dd (2 digits) Hour (1-12) |
247  * hh (2 digits) Hour (0-23) | HH (2 digits) Minute | mm (2 digits) Second | ss
248  * (2 digits) Author: Laurent Destailleur Author: Matelli (see
249  * http://matelli.fr/showcases/patchs-dolibarr/update-date-input-in-action-form.html)
250  * Licence: GPL
251  * ==================================================================
252  */
253 function formatDate(date,format)
254 {
255  // alert('formatDate date='+date+' format='+format);
256 
257  // Force parametres en chaine
258  format=format+"";
259 
260  var result="";
261 
262  var year=date.getYear()+""; if (year.length < 4) { year=""+(year-0+1900); }
263  var month=date.getMonth()+1;
264  var day=date.getDate();
265  var hour=date.getHours();
266  var minute=date.getMinutes();
267  var seconde=date.getSeconds();
268 
269  var i=0;
270  while (i < format.length)
271  {
272  c=format.charAt(i); // Recupere char du format
273  substr="";
274  j=i;
275  while ((format.charAt(j)==c) && (j < format.length)) // Recupere char successif identiques
276  {
277  substr += format.charAt(j++);
278  }
279 
280  // alert('substr='+substr);
281  if (substr == 'yyyy') { result=result+year; }
282  else if (substr == 'yy') { result=result+year.substring(2,4); }
283  else if (substr == 'M') { result=result+month; }
284  else if (substr == 'MM') { result=result+(month<1||month>9?"":"0")+month; }
285  else if (substr == 'd') { result=result+day; }
286  else if (substr == 'dd') { result=result+(day<1||day>9?"":"0")+day; }
287  else if (substr == 'hh') { if (hour > 12) hour-=12; result=result+(hour<0||hour>9?"":"0")+hour; }
288  else if (substr == 'HH') { result=result+(hour<0||hour>9?"":"0")+hour; }
289  else if (substr == 'mm') { result=result+(minute<0||minute>9?"":"0")+minute; }
290  else if (substr == 'ss') { result=result+(seconde<0||seconde>9?"":"0")+seconde; }
291  else { result=result+substr; }
292 
293  i+=substr.length;
294  }
295 
296  // alert(result);
297  return result;
298 }

301 /*
302  * =================================================================
303  * Function: getDateFromFormat(date_string, format_string)
304  * Purpose: This function takes a date string and a format string.
305  * It parses the date string with format and it
306  * returns the date as a javascript Date() object. If date does not match
307  * format, it returns 0. The format string can use the following tags:
308  * Field | Tags
309  * -------------+-----------------------------------
310  * Year | yyyy (4 digits), yy (2 digits)
311  * Month | MM (2 digits)
312  * Day of Month | dd (2 digits)
313  * Hour (1-12) | hh (2 digits)
314  * Hour (0-23) | HH (2 digits)
315  * Minute | mm (2 digits)
316  * Second | ss (2 digits)
317  * Author: Laurent Destailleur
318  * Licence: GPL
319  * ==================================================================
320  */
321 function getDateFromFormat(val,format)
322 {
323  // alert('getDateFromFormat val='+val+' format='+format);
324 
325  // Force parametres en chaine
326  val=val+"";
327  format=format+"";
328 
329  if (val == '') return 0;
330 
331  var now=new Date();
332  var year=now.getYear(); if (year.length < 4) { year=""+(year-0+1900); }
333  var month=now.getMonth()+1;
334  var day=now.getDate();
335  var hour=now.getHours();
336  var minute=now.getMinutes();
337  var seconde=now.getSeconds();
338 
339  var i=0;
340  var d=0; // -d- follows the date string while -i- follows the format
341  // string
342 
343  while (i < format.length)
344  {
345  c=format.charAt(i); // Recupere char du format
346  substr="";
347  j=i;
348  while ((format.charAt(j)==c) && (j < format.length)) // Recupere char
349  // successif
350  // identiques
351  {
352  substr += format.charAt(j++);
353  }
354 
355  // alert('substr='+substr);
356  if (substr == "yyyy") year=getIntegerInString(val,d,4,4);
357  if (substr == "yy") year=""+(getIntegerInString(val,d,2,2)-0+1900);
358  if (substr == "MM" ||substr == "M")
359  {
360  month=getIntegerInString(val,d,1,2);
361  if (month) d -= 2- month.length;
362  }
363  if (substr == "dd")
364  {
365  day=getIntegerInString(val,d,1,2);
366  if (day) d -= 2- day.length;
367  }
368  if (substr == "HH" ||substr == "hh" )
369  {
370  hour=getIntegerInString(val,d,1,2);
371  if (dhouray) d -= 2- hour.length;
372  }
373  if (substr == "mm"){
374  minute=getIntegerInString(val,d,1,2);
375  if (minute) d -= 2- minute.length;
376  }
377  if (substr == "ss")
378  {
379  seconde=getIntegerInString(val,d,1,2);
380  if (seconde) d -= 2- seconde.length;
381  }
382 
383  i+=substr.length;
384  d+=substr.length;
385  }
386 
387  // Check if format param are ok
388  if (year==null||year<1) { return 0; }
389  if (month==null||(month<1)||(month>12)) { return 0; }
390  if (day==null||(day<1)||(day>31)) { return 0; }
391  if (hour==null||(hour<0)||(hour>24)) { return 0; }
392  if (minute==null||(minute<0)||(minute>60)) { return 0; }
393  if (seconde==null||(seconde<0)||(seconde>60)) { return 0; }
394 
395  // alert(year+' '+month+' '+day+' '+hour+' '+minute+' '+seconde);
396  return new Date(year,month-1,day,hour,minute,seconde);
397 }

421  * Purpose: Return part of string from position i that is integer
422  * ==================================================================
423  */
424 function getIntegerInString(str,i,minlength,maxlength)
425 {
426  for (var x=maxlength; x>=minlength; x--)
427  {
428  var substr=str.substring(i,i+x);
429  if (substr.length < minlength) { return null; }
430  if (stringIsInteger(substr)) { return substr; }
431  }
432  return null;
