164  public function index($sortfield = "t.rowid", $sortorder = 'ASC', $limit = 100, $page = 0, $thirdparty_ids = '', $status = '', $sqlfilters = '')
165  {
166  global $db, $conf;
167 
168  if (!DolibarrApiAccess::$user->rights->facture->lire) {
169  throw new RestException(401);
170  }
171 
172  $obj_ret = array();
173 
174  // case of external user, $thirdparty_ids param is ignored and replaced by user's socid
175  $socids = DolibarrApiAccess::$user->socid ? DolibarrApiAccess::$user->socid : $thirdparty_ids;
176 
177  // If the internal user must only see his customers, force searching by him
178  $search_sale = 0;
179  if (!DolibarrApiAccess::$user->rights->societe->client->voir && !$socids) {
180  $search_sale = DolibarrApiAccess::$user->id;
181  }
182 
183  $sql = "SELECT t.rowid";
184  if ((!DolibarrApiAccess::$user->rights->societe->client->voir && !$socids) || $search_sale > 0) {
185  $sql .= ", sc.fk_soc, sc.fk_user"; // We need these fields in order to filter by sale (including the case where the user can only see his prospects)
186  }
187  $sql .= " FROM ".MAIN_DB_PREFIX."facture as t";
188 
189  if ((!DolibarrApiAccess::$user->rights->societe->client->voir && !$socids) || $search_sale > 0) {
190  $sql .= ", ".MAIN_DB_PREFIX."societe_commerciaux as sc"; // We need this table joined to the select in order to filter by sale
191  }
192 
193  $sql .= ' WHERE t.entity IN ('.getEntity('invoice').')';
194  if ((!DolibarrApiAccess::$user->rights->societe->client->voir && !$socids) || $search_sale > 0) {
195  $sql .= " AND t.fk_soc = sc.fk_soc";
196  }
197  if ($socids) {
198  $sql .= " AND t.fk_soc IN (".$this->db->sanitize($socids).")";
199  }
200 
201  if ($search_sale > 0) {
202  $sql .= " AND t.rowid = sc.fk_soc"; // Join for the needed table to filter by sale
203  }
204 
205  // Filter by status
206  if ($status == 'draft') {
207  $sql .= " AND t.fk_statut IN (0)";
208  }
209  if ($status == 'unpaid') {
210  $sql .= " AND t.fk_statut IN (1)";
211  }
212  if ($status == 'paid') {
213  $sql .= " AND t.fk_statut IN (2)";
214  }
215  if ($status == 'cancelled') {
216  $sql .= " AND t.fk_statut IN (3)";
217  }
218  // Insert sale filter
219  if ($search_sale > 0) {
220  $sql .= " AND sc.fk_user = ".((int) $search_sale);
221  }
222  // Add sql filters
223  if ($sqlfilters) {
224  if (!DolibarrApi::_checkFilters($sqlfilters)) {
225  throw new RestException(503, 'Error when validating parameter sqlfilters '.$sqlfilters);
226  }
227  $regexstring = '\(([^:\'\(\)]+:[^:\'\(\)]+:[^\(\)]+)\)';
228  $sql .= " AND (".preg_replace_callback('/'.$regexstring.'/', 'DolibarrApi::_forge_criteria_callback', $sqlfilters).")";
229  }
230 
231  $sql .= $this->db->order($sortfield, $sortorder);
232  if ($limit) {
233  if ($page < 0) {
234  $page = 0;
235  }
236  $offset = $limit * $page;
237 
238  $sql .= $this->db->plimit($limit + 1, $offset);
239  }
240 
241  $result = $this->db->query($sql);
242  if ($result) {
243  $i = 0;
244  $num = $this->db->num_rows($result);
245  $min = min($num, ($limit <= 0 ? $num : $limit));
246  while ($i < $min) {
247  $obj = $this->db->fetch_object($result);
248  $invoice_static = new Facture($this->db);
249  if ($invoice_static->fetch($obj->rowid)) {
250  // Get payment details
251  $invoice_static->totalpaid = $invoice_static->getSommePaiement();
252  $invoice_static->totalcreditnotes = $invoice_static->getSumCreditNotesUsed();
253  $invoice_static->totaldeposits = $invoice_static->getSumDepositsUsed();
254  $invoice_static->remaintopay = price2num($invoice_static->total_ttc - $invoice_static->totalpaid - $invoice_static->totalcreditnotes - $invoice_static->totaldeposits, 'MT');
255 
256  // Add external contacts ids
257  $invoice_static->contacts_ids = $invoice_static->liste_contact(-1, 'external', 1);
258 
259  $obj_ret[] = $this->_cleanObjectDatas($invoice_static);
260  }
261  $i++;
262  }
263  } else {
264  throw new RestException(503, 'Error when retrieve invoice list : '.$this->db->lasterror());
265  }
266  if (!count($obj_ret)) {
267  throw new RestException(404, 'No invoice found');
268  }
269  return $obj_ret;
270  }
