138  public static function convertSQLFromMysql($line, $type = 'ddl')
139  {
140  // Removed empty line if this is a comment line for SVN tagging
141  if (preg_match('/^--\s\$Id/i', $line)) {
142  return '';
143  }
144  // Return line if this is a comment
145  if (preg_match('/^#/i', $line) || preg_match('/^$/i', $line) || preg_match('/^--/i', $line)) {
146  return $line;
147  }
148  if ($line != "") {
149  if ($type == 'auto') {
150  if (preg_match('/ALTER TABLE/i', $line)) {
151  $type = 'dml';
152  } elseif (preg_match('/CREATE TABLE/i', $line)) {
153  $type = 'dml';
154  } elseif (preg_match('/DROP TABLE/i', $line)) {
155  $type = 'dml';
156  }
157  }
158 
159  if ($type == 'dml') {
160  $line = preg_replace('/\s/', ' ', $line); // Replace tabulation with space
161 
162  // we are inside create table statement so lets process datatypes
163  if (preg_match('/(ISAM|innodb)/i', $line)) { // end of create table sequence
164  $line = preg_replace('/\)[\s\t]*type[\s\t]*=[\s\t]*(MyISAM|innodb);/i', ');', $line);
165  $line = preg_replace('/\)[\s\t]*engine[\s\t]*=[\s\t]*(MyISAM|innodb);/i', ');', $line);
166  $line = preg_replace('/,$/', '', $line);
167  }
168 
169  // Process case: "CREATE TABLE llx_mytable(rowid integer NOT NULL AUTO_INCREMENT PRIMARY KEY,code..."
170  if (preg_match('/[\s\t\(]*(\w*)[\s\t]+int.*auto_increment/i', $line, $reg)) {
171  $newline = preg_replace('/([\s\t\(]*)([a-zA-Z_0-9]*)[\s\t]+int.*auto_increment[^,]*/i', '\\1 \\2 integer PRIMARY KEY AUTOINCREMENT', $line);
172  //$line = "-- ".$line." replaced by --\n".$newline;
173  $line = $newline;
174  }
175 
176  // tinyint type conversion
177  $line = str_replace('tinyint', 'smallint', $line);
178 
179  // nuke unsigned
180  $line = preg_replace('/(int\w+|smallint)\s+unsigned/i', '\\1', $line);
181 
182  // blob -> text
183  $line = preg_replace('/\w*blob/i', 'text', $line);
184 
185  // tinytext/mediumtext -> text
186  $line = preg_replace('/tinytext/i', 'text', $line);
187  $line = preg_replace('/mediumtext/i', 'text', $line);
188 
189  // change not null datetime field to null valid ones
190  // (to support remapping of "zero time" to null
191  $line = preg_replace('/datetime not null/i', 'datetime', $line);
192  $line = preg_replace('/datetime/i', 'timestamp', $line);
193 
194  // double -> numeric
195  $line = preg_replace('/^double/i', 'numeric', $line);
196  $line = preg_replace('/(\s*)double/i', '\\1numeric', $line);
197  // float -> numeric
198  $line = preg_replace('/^float/i', 'numeric', $line);
199  $line = preg_replace('/(\s*)float/i', '\\1numeric', $line);
200 
201  // unique index(field1,field2)
202  if (preg_match('/unique index\s*\((\w+\s*,\s*\w+)\)/i', $line)) {
203  $line = preg_replace('/unique index\s*\((\w+\s*,\s*\w+)\)/i', 'UNIQUE\(\\1\)', $line);
204  }
205 
206  // We remove end of requests "AFTER fieldxxx"
207  $line = preg_replace('/AFTER [a-z0-9_]+/i', '', $line);
208 
209  // We remove start of requests "ALTER TABLE tablexxx" if this is a DROP INDEX
210  $line = preg_replace('/ALTER TABLE [a-z0-9_]+ DROP INDEX/i', 'DROP INDEX', $line);
211 
212  // Translate order to rename fields
213  if (preg_match('/ALTER TABLE ([a-z0-9_]+) CHANGE(?: COLUMN)? ([a-z0-9_]+) ([a-z0-9_]+)(.*)$/i', $line, $reg)) {
214  $line = "-- ".$line." replaced by --\n";
215  $line .= "ALTER TABLE ".$reg[1]." RENAME COLUMN ".$reg[2]." TO ".$reg[3];
216  }
217 
218  // Translate order to modify field format
219  if (preg_match('/ALTER TABLE ([a-z0-9_]+) MODIFY(?: COLUMN)? ([a-z0-9_]+) (.*)$/i', $line, $reg)) {
220  $line = "-- ".$line." replaced by --\n";
221  $newreg3 = $reg[3];
222  $newreg3 = preg_replace('/ DEFAULT NULL/i', '', $newreg3);
223  $newreg3 = preg_replace('/ NOT NULL/i', '', $newreg3);
224  $newreg3 = preg_replace('/ NULL/i', '', $newreg3);
225  $newreg3 = preg_replace('/ DEFAULT 0/i', '', $newreg3);
226  $newreg3 = preg_replace('/ DEFAULT \'[0-9a-zA-Z_@]*\'/i', '', $newreg3);
227  $line .= "ALTER TABLE ".$reg[1]." ALTER COLUMN ".$reg[2]." TYPE ".$newreg3;
228  // TODO Add alter to set default value or null/not null if there is this in $reg[3]
229  }
230 
231  // alter table add primary key (field1, field2 ...) -> We create a unique index instead as dynamic creation of primary key is not supported
232  // ALTER TABLE llx_dolibarr_modules ADD PRIMARY KEY pk_dolibarr_modules (numero, entity);
233  if (preg_match('/ALTER\s+TABLE\s*(.*)\s*ADD\s+PRIMARY\s+KEY\s*(.*)\s*\((.*)$/i', $line, $reg)) {
234  $line = "-- ".$line." replaced by --\n";
235  $line .= "CREATE UNIQUE INDEX ".$reg[2]." ON ".$reg[1]."(".$reg[3];
236  }
237 
238  // Translate order to drop foreign keys
239  // ALTER TABLE llx_dolibarr_modules DROP FOREIGN KEY fk_xxx;
240  if (preg_match('/ALTER\s+TABLE\s*(.*)\s*DROP\s+FOREIGN\s+KEY\s*(.*)$/i', $line, $reg)) {
241  $line = "-- ".$line." replaced by --\n";
242  $line .= "ALTER TABLE ".$reg[1]." DROP CONSTRAINT ".$reg[2];
243  }
244 
245  // alter table add [unique] [index] (field1, field2 ...)
246  // ALTER TABLE llx_accountingaccount ADD INDEX idx_accountingaccount_fk_pcg_version (fk_pcg_version)
247  if (preg_match('/ALTER\s+TABLE\s*(.*)\s*ADD\s+(UNIQUE INDEX|INDEX|UNIQUE)\s+(.*)\s*\(([\w,\s]+)\)/i', $line, $reg)) {
248  $fieldlist = $reg[4];
249  $idxname = $reg[3];
250  $tablename = $reg[1];
251  $line = "-- ".$line." replaced by --\n";
252  $line .= "CREATE ".(preg_match('/UNIQUE/', $reg[2]) ? 'UNIQUE ' : '')."INDEX ".$idxname." ON ".$tablename." (".$fieldlist.")";
253  }
254  if (preg_match('/ALTER\s+TABLE\s*(.*)\s*ADD\s+CONSTRAINT\s+(.*)\s*FOREIGN\s+KEY\s*\(([\w,\s]+)\)\s*REFERENCES\s+(\w+)\s*\(([\w,\s]+)\)/i', $line, $reg)) {
255  // Pour l'instant les contraintes ne sont pas créées
256  dol_syslog(get_class().'::query line emptied');
257  $line = 'SELECT 0;';
258  }
259 
260  //if (preg_match('/rowid\s+.*\s+PRIMARY\s+KEY,/i', $line)) {
261  //preg_replace('/(rowid\s+.*\s+PRIMARY\s+KEY\s*,)/i', '/* \\1 */', $line);
262  //}
263  }
264 
265  // Delete using criteria on other table must not declare twice the deleted table
266  // DELETE FROM tabletodelete USING tabletodelete, othertable -> DELETE FROM tabletodelete USING othertable
267  if (preg_match('/DELETE FROM ([a-z_]+) USING ([a-z_]+), ([a-z_]+)/i', $line, $reg)) {
268  if ($reg[1] == $reg[2]) { // If same table, we remove second one
269  $line = preg_replace('/DELETE FROM ([a-z_]+) USING ([a-z_]+), ([a-z_]+)/i', 'DELETE FROM \\1 USING \\3', $line);
270  }
271  }
272 
273  // Remove () in the tables in FROM if one table
274  $line = preg_replace('/FROM\s*\((([a-z_]+)\s+as\s+([a-z_]+)\s*)\)/i', 'FROM \\1', $line);
275  //print $line."\n";
276 
277  // Remove () in the tables in FROM if two table
278  $line = preg_replace('/FROM\s*\(([a-z_]+\s+as\s+[a-z_]+)\s*,\s*([a-z_]+\s+as\s+[a-z_]+\s*)\)/i', 'FROM \\1, \\2', $line);
279  //print $line."\n";
280 
281  // Remove () in the tables in FROM if two table
282  $line = preg_replace('/FROM\s*\(([a-z_]+\s+as\s+[a-z_]+)\s*,\s*([a-z_]+\s+as\s+[a-z_]+\s*),\s*([a-z_]+\s+as\s+[a-z_]+\s*)\)/i', 'FROM \\1, \\2, \\3', $line);
283  //print $line."\n";
284 
285  //print "type=".$type." newline=".$line."<br>\n";
286  }
287 
288  return $line;
289  }

399  public function query($query, $usesavepoint = 0, $type = 'auto', $result_mode = 0)
400  {
401  global $conf, $dolibarr_main_db_readonly;
402 
403  $ret = null;
404 
405  $query = trim($query);
406 
407  $this->error = '';
408 
409  // Convert MySQL syntax to SQLite syntax
410  if (preg_match('/ALTER\s+TABLE\s*(.*)\s*ADD\s+CONSTRAINT\s+(.*)\s*FOREIGN\s+KEY\s*\(([\w,\s]+)\)\s*REFERENCES\s+(\w+)\s*\(([\w,\s]+)\)/i', $query, $reg)) {
411  // Ajout d'une clef étrangère à la table
412  // procédure de remplacement de la table pour ajouter la contrainte
413  // Exemple : ALTER TABLE llx_adherent ADD CONSTRAINT adherent_fk_soc FOREIGN KEY (fk_soc) REFERENCES llx_societe (rowid)
414  // -> CREATE TABLE ( ... ,CONSTRAINT adherent_fk_soc FOREIGN KEY (fk_soc) REFERENCES llx_societe (rowid))
415  $foreignFields = $reg[5];
416  $foreignTable = $reg[4];
417  $localfields = $reg[3];
418  $constraintname = trim($reg[2]);
419  $tablename = trim($reg[1]);
420 
421  $descTable = $this->db->querySingle("SELECT sql FROM sqlite_master WHERE name='".$this->escape($tablename)."'");
422 
423  // 1- Renommer la table avec un nom temporaire
424  $this->query("ALTER TABLE ".$tablename." RENAME TO tmp_".$tablename);
425 
426  // 2- Recréer la table avec la contrainte ajoutée
427 
428  // on bricole la requete pour ajouter la contrainte
429  $descTable = substr($descTable, 0, strlen($descTable) - 1);
430  $descTable .= ", CONSTRAINT ".$constraintname." FOREIGN KEY (".$localfields.") REFERENCES ".$foreignTable."(".$foreignFields.")";
431 
432  // fermeture de l'instruction
433  $descTable .= ')';
434 
435  // Création proprement dite de la table
436  $this->query($descTable);
437 
438  // 3- Transférer les données
439  $this->query("INSERT INTO ".$tablename." SELECT * FROM tmp_".$tablename);
440 
441  // 4- Supprimer la table temporaire
442  $this->query("DROP TABLE tmp_".$tablename);
443 
444  // dummy statement
445  $query = "SELECT 0";
446  } else {
447  $query = $this->convertSQLFromMysql($query, $type);
448  }
449  //print "After convertSQLFromMysql:\n".$query."<br>\n";
450 
451  if (!in_array($query, array('BEGIN', 'COMMIT', 'ROLLBACK'))) {
452  $SYSLOG_SQL_LIMIT = 10000; // limit log to 10kb per line to limit DOS attacks
453  dol_syslog('sql='.substr($query, 0, $SYSLOG_SQL_LIMIT), LOG_DEBUG);
454  }
455  if (empty($query)) {
456  return false; // Return false = error if empty request
457  }
458 
459  if (!empty($dolibarr_main_db_readonly)) {
460  if (preg_match('/^(INSERT|UPDATE|REPLACE|DELETE|CREATE|ALTER|TRUNCATE|DROP)/i', $query)) {
461  $this->lasterror = 'Application in read-only mode';
462  $this->lasterrno = 'APPREADONLY';
463  $this->lastquery = $query;
464  return false;
465  }
466  }
467 
468  // Ordre SQL ne necessitant pas de connexion a une base (exemple: CREATE DATABASE)
469  try {
470  //$ret = $this->db->exec($query);
471  $ret = $this->db->query($query); // $ret is a Sqlite3Result
472  if ($ret) {
473  $ret->queryString = $query;
474  }
475  } catch (Exception $e) {
476  $this->error = $this->db->lastErrorMsg();
477  }
478 
479  if (!preg_match("/^COMMIT/i", $query) && !preg_match("/^ROLLBACK/i", $query)) {
480  // Si requete utilisateur, on la sauvegarde ainsi que son resultset
481  if (!is_object($ret) || $this->error) {
482  $this->lastqueryerror = $query;
483  $this->lasterror = $this->error();
484  $this->lasterrno = $this->errno();
485 
486  dol_syslog(get_class($this)."::query SQL Error query: ".$query, LOG_ERR);
487 
488  $errormsg = get_class($this)."::query SQL Error message: ".$this->lasterror;
489 
490  if (preg_match('/[0-9]/', $this->lasterrno)) {
491  $errormsg .= ' ('.$this->lasterrno.')';
492  }
493 
494  if ($conf->global->SYSLOG_LEVEL < LOG_DEBUG) {
495  dol_syslog(get_class($this)."::query SQL Error query: ".$query, LOG_ERR); // Log of request was not yet done previously
496  }
497  dol_syslog(get_class($this)."::query SQL Error message: ".$errormsg, LOG_ERR);
498  }
499  $this->lastquery = $query;
500  $this->_results = $ret;
501  }
502 
503  return $ret;
504  }

665  {
