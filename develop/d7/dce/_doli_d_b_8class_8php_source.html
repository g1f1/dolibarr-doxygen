239  public function order($sortfield = null, $sortorder = null)
240  {
241  if (!empty($sortfield)) {
242  $oldsortorder = '';
243  $return = '';
244  $fields = explode(',', $sortfield);
245  $orders = explode(',', $sortorder);
246  $i = 0;
247  foreach ($fields as $val) {
248  if (!$return) {
249  $return .= ' ORDER BY ';
250  } else {
251  $return .= ', ';
252  }
253 
254  $return .= preg_replace('/[^0-9a-z_\.]/i', '', $val); // Add field
255 
256  $tmpsortorder = (empty($orders[$i]) ? '' : trim($orders[$i]));
257 
258  // Only ASC and DESC values are valid SQL
259  if (strtoupper($tmpsortorder) === 'ASC') {
260  $oldsortorder = 'ASC';
261  $return .= ' ASC';
262  } elseif (strtoupper($tmpsortorder) === 'DESC') {
263  $oldsortorder = 'DESC';
264  $return .= ' DESC';
265  } else {
266  $return .= ' '.($oldsortorder ? $oldsortorder : 'ASC');
267  }
268 
269  $i++;
270  }
271  return $return;
272  } else {
273  return '';
274  }
