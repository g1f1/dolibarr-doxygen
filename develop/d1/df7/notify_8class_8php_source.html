127  } elseif ($nb >= 2) {
128  $texte = img_object($langs->trans("Notifications"), 'email').' '.$langs->trans("SomeNotificationsWillBeSent", $nb);

130

132  $i = 0;
133  foreach ($listofnotiftodo as $val) {
134  if ($i) {
135  $texte .= ', ';
136  } else {
137  $texte .= ' (';

142  $texte .= $val['emaildesc'];
143  }

146  if ($i) {

303  }
304  }
305  }
306  }
307  }

309  if ($error) {
310  return -1;
311  }
312
