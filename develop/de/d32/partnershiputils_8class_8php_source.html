207  return ($error ? 1 : 0);

399  private function checkDolibarrBacklink($website = null)
400  {
401  global $conf, $langs, $user;
402 
403  $found = 0;
404  $error = 0;
405  $webcontent = '';
406 
407  // $website = 'https://nextgestion.com/'; // For Test
408  $tmpgeturl = getURLContent($website, 'GET', '', 1, array(), array('http', 'https'), 0);
409  if ($tmpgeturl['curl_error_no']) {
410  $error++;
411  dol_syslog('Error getting '.$website.': '.$tmpgeturl['curl_error_msg']);
412  } elseif ($tmpgeturl['http_code'] != '200') {
413  $error++;
414  dol_syslog('Error getting '.$website.': '.$tmpgeturl['curl_error_msg']);
415  } else {
416  $urlContent = $tmpgeturl['content'];
417  $dom = new DOMDocument();
418  @$dom->loadHTML($urlContent);
419 
420  $xpath = new DOMXPath($dom);
421  $hrefs = $xpath->evaluate("//a");
422 
423  for ($i = 0; $i < $hrefs->length; $i++) {
424  $href = $hrefs->item($i);
425  $url = $href->getAttribute('href');
426  $url = filter_var($url, FILTER_SANITIZE_URL);
427  if (!filter_var($url, FILTER_VALIDATE_URL) === false) {
428  $webcontent .= $url;
429  }
430  }
431  }
432 
433  if ($webcontent && !empty($conf->global->PARTNERSHIP_BACKLINKS_TO_CHECK) && preg_match('/'.$conf->global->PARTNERSHIP_BACKLINKS_TO_CHECK.'/', $webcontent)) {
434  $found = 1;
435  }
436 
437  return $found;
438  }
