102  public function sendfile()
103  {
104  global $conf;
105 
106  $errorlevel = error_reporting();
107  error_reporting($errorlevel ^ E_WARNING); // Desactive warnings
108 
109  $res = false;
110 
111  dol_syslog("CSMSFile::sendfile addr_to=".$this->addr_to, LOG_DEBUG);
112  dol_syslog("CSMSFile::sendfile message=\n".$this->message);
113 
114  $this->message = stripslashes($this->message);
115 
116  if (!empty($conf->global->MAIN_SMS_DEBUG)) $this->dump_sms();
117 
118  if (empty($conf->global->MAIN_DISABLE_ALL_SMS))
119  {
120  // Action according to choosed sending method
121  if ($conf->global->MAIN_SMS_SENDMODE == 'ovh') // Backward compatibility @deprecated
122  {
123  dol_include_once('/ovh/class/ovhsms.class.php');
124  $sms = new OvhSms($this->db);
125  $sms->expe = $this->addr_from;
126  $sms->dest = $this->addr_to;
127  $sms->message = $this->message;
128  $sms->deferred = $this->deferred;
129  $sms->priority = $this->priority;
130  $sms->class = $this->class;
131  $sms->nostop = $this->nostop;
132 
133  $sms->socid = $this->socid;
134  $sms->contact_id = $this->contact_id;
135  $sms->project = $this->fk_project;
136 
137  $res = $sms->SmsSend();
138 
139  if ($res <= 0)
140  {
141  $this->error = $sms->error;
142  dol_syslog("CSMSFile::sendfile: sms send error=".$this->error, LOG_ERR);
143  } else {
144  dol_syslog("CSMSFile::sendfile: sms send success with id=".$res, LOG_DEBUG);
145  //var_dump($res); // 1973128
146  if (!empty($conf->global->MAIN_SMS_DEBUG)) $this->dump_sms_result($res);
147  }
148  } elseif (!empty($conf->global->MAIN_SMS_SENDMODE)) // $conf->global->MAIN_SMS_SENDMODE looks like a value 'class@module'
149  {
150  $tmp = explode('@', $conf->global->MAIN_SMS_SENDMODE);
151  $classfile = $tmp[0]; $module = (empty($tmp[1]) ? $tmp[0] : $tmp[1]);
152  dol_include_once('/'.$module.'/class/'.$classfile.'.class.php');
153  try {
154  $classname = ucfirst($classfile);
155  $sms = new $classname($this->db);
156  $sms->expe = $this->addr_from;
157  $sms->dest = $this->addr_to;
158  $sms->deferred = $this->deferred;
159  $sms->priority = $this->priority;
160  $sms->class = $this->class;
161  $sms->message = $this->message;
162  $sms->nostop = $this->nostop;
163 
164  $sms->socid = $this->socid;
165  $sms->contact_id = $this->contact_id;
166  $sms->fk_project = $this->fk_project;
167 
168  $res = $sms->SmsSend();
169 
170  $this->error = $sms->error;
171  $this->errors = $sms->errors;
172  if ($res <= 0)
173  {
174  dol_syslog("CSMSFile::sendfile: sms send error=".$this->error, LOG_ERR);
175  } else {
176  dol_syslog("CSMSFile::sendfile: sms send success with id=".$res, LOG_DEBUG);
177  //var_dump($res); // 1973128
178  if (!empty($conf->global->MAIN_SMS_DEBUG)) $this->dump_sms_result($res);
179  }
180  } catch (Exception $e)
181  {
182  dol_print_error('', 'Error to get list of senders: '.$e->getMessage());
183  }
184  } else {
185  // Send sms method not correctly defined
186  // --------------------------------------
187 
188  return 'Bad value for MAIN_SMS_SENDMODE constant';
189  }
190  } else {
191  $this->error = 'No sms sent. Feature is disabled by option MAIN_DISABLE_ALL_SMS';
192  dol_syslog("CSMSFile::sendfile: ".$this->error, LOG_WARNING);
193  }
194 
195  error_reporting($errorlevel); // Reactive niveau erreur origine
196 
197  return $res;
198  }
