252 function getDefaultDatesForTransfer()
253 {
254  global $db, $conf;
255 
256  // Period by default on transfer (0: previous month | 1: current month | 2: fiscal year)
257  $periodbydefaultontransfer = $conf->global->ACCOUNTING_DEFAULT_PERIOD_ON_TRANSFER;
258  isset($periodbydefaultontransfer) ? $periodbydefaultontransfer : 0;
259  if ($periodbydefaultontransfer == 2) {
260  $sql = "SELECT date_start, date_end FROM ".MAIN_DB_PREFIX."accounting_fiscalyear ";
261  $sql .= " WHERE date_start < '".$db->idate(dol_now())."' AND date_end > '".$db->idate(dol_now())."'";
262  $sql .= $db->plimit(1);
263  $res = $db->query($sql);
264  if ($res->num_rows > 0) {
265  $fiscalYear = $db->fetch_object($res);
266  $date_start = strtotime($fiscalYear->date_start);
267  $date_end = strtotime($fiscalYear->date_end);
268  } else {
269  $month_start = ($conf->global->SOCIETE_FISCAL_MONTH_START ? ($conf->global->SOCIETE_FISCAL_MONTH_START) : 1);
270  $year_start = dol_print_date(dol_now(), '%Y');
271  if ($conf->global->SOCIETE_FISCAL_MONTH_START > dol_print_date(dol_now(), '%m')) {
272  $year_start = $year_start - 1;
273  }
274  $year_end = $year_start + 1;
275  $month_end = $month_start - 1;
276  if ($month_end < 1)
277  {
278  $month_end = 12;
279  $year_end--;
280  }
281  $date_start = dol_mktime(0, 0, 0, $month_start, 1, $year_start);
282  $date_end = dol_get_last_day($year_end, $month_end);
283  }
284  } elseif ($periodbydefaultontransfer == 1) {
285  $year_current = strftime("%Y", dol_now());
286  $pastmonth = strftime("%m", dol_now());
287  $pastmonthyear = $year_current;
288  if ($pastmonth == 0) {
289  $pastmonth = 12;
290  $pastmonthyear--;
291  }
292  } else {
293  $year_current = strftime("%Y", dol_now());
294  $pastmonth = strftime("%m", dol_now()) - 1;
295  $pastmonthyear = $year_current;
296  if ($pastmonth == 0) {
297  $pastmonth = 12;
298  $pastmonthyear--;
299  }
300  }
301 
302  return array(
303  'date_start' => $date_start,
304  'date_end' => $date_end,
305  'pastmonthyear' => $pastmonthyear,
306  'pastmonth' => $pastmonth
307  );
308 }
