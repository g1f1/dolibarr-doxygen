220  $result = $object->createCommon($user);

222  $error++;
223  $this->error = $object->error;
224  $this->errors = $object->errors;
225  }
226 
227  unset($object->context['createfromclone']);
228 
229  // End
230  if (!$error) {
231  $this->db->commit();
232  return $object;

234  $this->db->rollback();
