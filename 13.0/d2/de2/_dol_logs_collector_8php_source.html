139  protected function tailFile($file, $lines)
140  {
141  $handle = fopen($file, "r");
142  $linecounter = $lines;
143  $pos = -2;
144  $beginning = false;
145  $text = array();
146  while ($linecounter > 0) {
147  $t = " ";
148  while ($t != "\n") {
149  if (fseek($handle, $pos, SEEK_END) == -1) {
150  $beginning = true;
151  break;
152  }
153  $t = fgetc($handle);
154  $pos--;
155  }
156  $linecounter--;
157  if ($beginning) {
158  rewind($handle);
159  }
160  $text[$lines - $linecounter - 1] = fgets($handle);
161  if ($beginning) {
162  break;
163  }
164  }
165  fclose($handle);
166  return array_reverse($text);
