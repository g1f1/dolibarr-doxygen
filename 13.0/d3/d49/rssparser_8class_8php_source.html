508  public function feed_start_element($p, $element, &$attrs)
509  {
510  // phpcs:enable
511  $el = $element = strtolower($element);
512  $attrs = array_change_key_case($attrs, CASE_LOWER);
513 
514  // check for a namespace, and split if found
515  $ns = false;
516  if (strpos($element, ':')) {
517  list($ns, $el) = explode(':', $element, 2);
518  }
519  if ($ns and $ns != 'rdf') {
520  $this->current_namespace = $ns;
521  }
522 
523  // if feed type isn't set, then this is first element of feed identify feed from root element
524  if (empty($this->_format)) {
525  if ($el == 'rdf') {
526  $this->_format = 'rss';
527  $this->feed_version = '1.0';
528  } elseif ($el == 'rss') {
529  $this->_format = 'rss';
530  $this->feed_version = $attrs['version'];
531  } elseif ($el == 'feed') {
532  $this->_format = 'atom';
533  $this->feed_version = $attrs['version'];
534  $this->inchannel = true;
535  }
536  return;
537  }
538 
539  if ($el == 'channel') {
540  $this->inchannel = true;
541  } elseif ($el == 'item' or $el == 'entry') {
542  $this->initem = true;
543  if (isset($attrs['rdf:about'])) {
544  $this->current_item['about'] = $attrs['rdf:about'];
545  }
546  } elseif ($this->_format == 'rss' and
547  $this->current_namespace == '' and
548  $el == 'textinput') {
549  // if we're in the default namespace of an RSS feed,
550  // record textinput or image fields
551  $this->intextinput = true;
552  } elseif ($this->_format == 'rss' and
553  $this->current_namespace == '' and
554  $el == 'image') {
555  $this->inimage = true;
556  } elseif ($this->_format == 'atom' and in_array($el, $this->_CONTENT_CONSTRUCTS)) {
557  // handle atom content constructs
558  // avoid clashing w/ RSS mod_content
559  if ($el == 'content') {
560  $el = 'atom_content';
561  }
562 
563  $this->incontent = $el;
564  } elseif ($this->_format == 'atom' and $this->incontent) {
565  // if inside an Atom content construct (e.g. content or summary) field treat tags as text
566  // if tags are inlined, then flatten
567  $attrs_str = join(' ', array_map('map_attrs', array_keys($attrs), array_values($attrs)));
568 
569  $this->append_content("<$element $attrs_str>");
570 
571  array_unshift($this->stack, $el);
572  } elseif ($this->_format == 'atom' and $el == 'link') {
573  // Atom support many links per containging element.
574  // Magpie treats link elements of type rel='alternate'
575  // as being equivalent to RSS's simple link element.
576  if (isset($attrs['rel']) && $attrs['rel'] == 'alternate') {
577  $link_el = 'link';
578  } elseif (!isset($attrs['rel'])) {
579  $link_el = 'link';
580  } else {
581  $link_el = 'link_'.$attrs['rel'];
582  }
583 
584  $this->append($link_el, $attrs['href']);
585  } else {
586  // set stack[0] to current element
587  array_unshift($this->stack, $el);
588  }
589  }

817  $array[$key] = $child;
818  }
