88  protected function checkStatusCode($status_code)
89  {
90  $error_label = 'This call to PrestaShop Web Services failed and returned an HTTP status of %d. That means: %s.';
91  switch ($status_code)
92  {
93  case 200:
94  case 201:
95  break;
96  case 204:
97  throw new PrestaShopWebserviceException(sprintf($error_label, $status_code, 'No content'));
98  case 400:
99  throw new PrestaShopWebserviceException(sprintf($error_label, $status_code, 'Bad Request'));
100  case 401:
101  throw new PrestaShopWebserviceException(sprintf($error_label, $status_code, 'Unauthorized'));
102  case 404:
103  throw new PrestaShopWebserviceException(sprintf($error_label, $status_code, 'Not Found'));
104  case 405:
105  throw new PrestaShopWebserviceException(sprintf($error_label, $status_code, 'Method Not Allowed'));
106  case 500:
107  throw new PrestaShopWebserviceException(sprintf($error_label, $status_code, 'Internal Server Error'));
108  default:
109  throw new PrestaShopWebserviceException('This call to PrestaShop Web Services returned an unexpected HTTP status of:'.$status_code);
110  }
111  }

120  public function executeRequest($url, $curl_params = array())
121  {
122  $defaultParams = array(
123  CURLOPT_HEADER => true,
124  CURLOPT_RETURNTRANSFER => true,
125  CURLINFO_HEADER_OUT => true,
126  CURLOPT_HTTPAUTH => CURLAUTH_BASIC,
127  CURLOPT_USERPWD => $this->key.':',
128  CURLOPT_HTTPHEADER => array('Expect:')
129  );
130 
131  dol_syslog("curl_init url=".$url);
132  $session = curl_init($url);
133 
134  $curl_options = array();
135  foreach ($defaultParams as $defkey => $defval)
136  {
137  if (isset($curl_params[$defkey]))
138  $curl_options[$defkey] = $curl_params[$defkey];
139  else {
140  $curl_options[$defkey] = $defaultParams[$defkey];
141  }
142  }
143  foreach ($curl_params as $defkey => $defval)
144  if (!isset($curl_options[$defkey]))
145  $curl_options[$defkey] = $curl_params[$defkey];
146 
147  dol_syslog("curl curl_options = ".var_export($curl_options, true));
148  curl_setopt_array($session, $curl_options);
149  $response = curl_exec($session);
150 
151  $index = strpos($response, "\r\n\r\n");
152  if ($index === false && $curl_params[CURLOPT_CUSTOMREQUEST] != 'HEAD')
153  throw new PrestaShopWebserviceException('Bad HTTP response');
154 
155  $header = substr($response, 0, $index);
156  $body = substr($response, $index + 4);
157 
158  $headerArrayTmp = explode("\n", $header);
159 
160  $headerArray = array();
161  foreach ($headerArrayTmp as &$headerItem)
162  {
163  $tmp = explode(':', $headerItem);
164  $tmp = array_map('trim', $tmp);
165  if (count($tmp) == 2)
166  $headerArray[$tmp[0]] = $tmp[1];
167  }
168 
169  if (array_key_exists('PSWS-Version', $headerArray))
170  {
171  $this->version = $headerArray['PSWS-Version'];
172  if (
173  version_compare(PrestaShopWebservice::PSCOMPATIBLEVERSIONMIN, $headerArray['PSWS-Version']) == 1 ||
174  version_compare(PrestaShopWebservice::PSCOMPATIBLEVERSIONMAX, $headerArray['PSWS-Version']) == -1
175  )
176  throw new PrestaShopWebserviceException('This library is not compatible with this version of PrestaShop. Please upgrade/downgrade this library');
177  }
178 
179  if ($this->debug)
180  {
181  $this->printDebug('HTTP REQUEST HEADER', curl_getinfo($session, CURLINFO_HEADER_OUT));
182  $this->printDebug('HTTP RESPONSE HEADER', $header);
183  }
184  $status_code = curl_getinfo($session, CURLINFO_HTTP_CODE);
185  if ($status_code === 0)
186  throw new PrestaShopWebserviceException('CURL Error: '.curl_error($session));
187  curl_close($session);
188  if ($this->debug)
189  {
190  if ($curl_params[CURLOPT_CUSTOMREQUEST] == 'PUT' || $curl_params[CURLOPT_CUSTOMREQUEST] == 'POST')
191  $this->printDebug('XML SENT', urldecode($curl_params[CURLOPT_POSTFIELDS]));
192  if ($curl_params[CURLOPT_CUSTOMREQUEST] != 'DELETE' && $curl_params[CURLOPT_CUSTOMREQUEST] != 'HEAD')
193  $this->printDebug('RETURN HTTP BODY', $body);
194  }
195  return array('status_code' => $status_code, 'response' => $body, 'header' => $header);
196  }

231  {
232  libxml_clear_errors();
233  libxml_use_internal_errors(true);
234  $xml = simplexml_load_string($response, 'SimpleXMLElement', LIBXML_NOCDATA);
235  if (libxml_get_errors())

237  $msg = var_export(libxml_get_errors(), true);
238  libxml_clear_errors();
239  throw new PrestaShopWebserviceException('HTTP XML response is not parsable: '.$msg);

243  throw new PrestaShopWebserviceException('HTTP response is empty');
