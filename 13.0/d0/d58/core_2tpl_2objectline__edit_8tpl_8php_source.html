347  function checkFreeLine(e, npRate)
348  {
349  var buying_price = $("input[name='buying_price']:first");
350  var remise = $("input[name='remise_percent']:first");
351 
352  var rate = $("input[name='"+npRate+"']:first");

354  return true;
355 
356  var ratejs = price2numjs(rate.val());
357  if (! $.isNumeric(ratejs))
358  {
359  alert('<?php echo dol_escape_js($langs->transnoentities("rateMustBeNumeric")); ?>');
360  e.stopPropagation();
361  setTimeout(function () { rate.focus() }, 50);
362  return false;
363  }
364  if (npRate == "np_markRate" && rate.val() >= 100)
365  {
366  alert('<?php echo dol_escape_js($langs->transnoentities("markRateShouldBeLesserThan100")); ?>');
367  e.stopPropagation();
368  setTimeout(function () { rate.focus() }, 50);
369  return false;
370  }
371 
372  var price = 0;
373  remisejs=price2numjs(remise.val());
374 
375  if (remisejs != 100) // If a discount not 100 or no discount
376  {
377  if (remisejs == '') remisejs=0;
378 
379  bpjs=price2numjs(buying_price.val());
380  ratejs=price2numjs(rate.val());
381 
382  if (npRate == "np_marginRate")
383  price = ((bpjs * (1 + ratejs / 100)) / (1 - remisejs / 100));
384  else if (npRate == "np_markRate")
385  price = ((bpjs / (1 - ratejs / 100)) / (1 - remisejs / 100));
386  }
387  $("input[name='price_ht']:first").val(price); // TODO Must use a function like php price to have here a formated value
