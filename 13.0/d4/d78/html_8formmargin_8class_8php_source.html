62  public function getMarginInfosArray($object, $force_price = false)
63  {
64  global $conf, $db;
65 
66  // Default returned array
67  $marginInfos = array(
68  'pa_products' => 0,
69  'pv_products' => 0,
70  'margin_on_products' => 0,
71  'margin_rate_products' => '',
72  'mark_rate_products' => '',
73  'pa_services' => 0,
74  'pv_services' => 0,
75  'margin_on_services' => 0,
76  'margin_rate_services' => '',
77  'mark_rate_services' => '',
78  'pa_total' => 0,
79  'pv_total' => 0,
80  'total_margin' => 0,
81  'total_margin_rate' => '',
82  'total_mark_rate' => ''
83  );
84 
85  foreach ($object->lines as $line)
86  {
87  if (empty($line->pa_ht) && isset($line->fk_fournprice) && !$force_price)
88  {
89  require_once DOL_DOCUMENT_ROOT.'/fourn/class/fournisseur.product.class.php';
90  $product = new ProductFournisseur($this->db);
91  if ($product->fetch_product_fournisseur_price($line->fk_fournprice))
92  $line->pa_ht = $product->fourn_unitprice * (1 - $product->fourn_remise_percent / 100);
93  }
94 
95  // If buy price is not defined (null), we will use the sell price. If defined to 0 (it means it was forced to 0 during insert, for example for a free to get product), we must still use 0.
96  //if ((!isset($line->pa_ht) || $line->pa_ht == 0) && $line->subprice > 0 && (isset($conf->global->ForceBuyingPriceIfNull) && $conf->global->ForceBuyingPriceIfNull > 0)) {
97  if ((!isset($line->pa_ht)) && $line->subprice > 0 && (isset($conf->global->ForceBuyingPriceIfNull) && $conf->global->ForceBuyingPriceIfNull > 0)) {
98  $line->pa_ht = $line->subprice * (1 - ($line->remise_percent / 100));
99  }
100 
101  $pv = $line->total_ht;
102  $pa_ht = ($pv < 0 ? -$line->pa_ht : $line->pa_ht); // We choosed to have line->pa_ht always positive in database, so we guess the correct sign
103  if (($object->element == 'facture' && $object->type == $object::TYPE_SITUATION)
104  || ($object->element == 'facture' && $object->type == $object::TYPE_CREDIT_NOTE && $conf->global->INVOICE_USE_SITUATION_CREDIT_NOTE && $object->situation_counter > 0)) {
105  $pa = $line->qty * $pa_ht * ($line->situation_percent / 100);
106  } else {
107  $pa = $line->qty * $pa_ht;
108  }
109 
110  // calcul des marges
111  if (isset($line->fk_remise_except) && isset($conf->global->MARGIN_METHODE_FOR_DISCOUNT)) { // remise
112  if ($conf->global->MARGIN_METHODE_FOR_DISCOUNT == '1') { // remise globale considérée comme produit
113  $marginInfos['pa_products'] += $pa;
114  $marginInfos['pv_products'] += $pv;
115  $marginInfos['pa_total'] += $pa;
116  $marginInfos['pv_total'] += $pv;
117  // if credit note, margin = -1 * (abs(selling_price) - buying_price)
118  //if ($pv < 0)
119  //{
120  // $marginInfos['margin_on_products'] += -1 * (abs($pv) - $pa);
121  //}
122  //else
123  $marginInfos['margin_on_products'] += $pv - $pa;
124  } elseif ($conf->global->MARGIN_METHODE_FOR_DISCOUNT == '2') { // remise globale considérée comme service
125  $marginInfos['pa_services'] += $pa;
126  $marginInfos['pv_services'] += $pv;
127  $marginInfos['pa_total'] += $pa;
128  $marginInfos['pv_total'] += $pv;
129  // if credit note, margin = -1 * (abs(selling_price) - buying_price)
130  //if ($pv < 0)
131  // $marginInfos['margin_on_services'] += -1 * (abs($pv) - $pa);
132  //else
133  $marginInfos['margin_on_services'] += $pv - $pa;
134  } elseif ($conf->global->MARGIN_METHODE_FOR_DISCOUNT == '3') { // remise globale prise en compte uniqt sur total
135  $marginInfos['pa_total'] += $pa;
136  $marginInfos['pv_total'] += $pv;
137  }
138  } else {
139  $type = $line->product_type ? $line->product_type : $line->fk_product_type;
140  if ($type == 0) { // product
141  $marginInfos['pa_products'] += $pa;
142  $marginInfos['pv_products'] += $pv;
143  $marginInfos['pa_total'] += $pa;
144  $marginInfos['pv_total'] += $pv;
145  // if credit note, margin = -1 * (abs(selling_price) - buying_price)
146  //if ($pv < 0)
147  //{
148  // $marginInfos['margin_on_products'] += -1 * (abs($pv) - $pa);
149  //}
150  //else
151  //{
152  $marginInfos['margin_on_products'] += $pv - $pa;
153  //}
154  } elseif ($type == 1) { // service
155  $marginInfos['pa_services'] += $pa;
156  $marginInfos['pv_services'] += $pv;
157  $marginInfos['pa_total'] += $pa;
158  $marginInfos['pv_total'] += $pv;
159  // if credit note, margin = -1 * (abs(selling_price) - buying_price)
160  //if ($pv < 0)
161  // $marginInfos['margin_on_services'] += -1 * (abs($pv) - $pa);
162  //else
163  $marginInfos['margin_on_services'] += $pv - $pa;
164  }
165  }
166  }
167  if ($marginInfos['pa_products'] > 0)
168  $marginInfos['margin_rate_products'] = 100 * $marginInfos['margin_on_products'] / $marginInfos['pa_products'];
169  if ($marginInfos['pv_products'] > 0)
170  $marginInfos['mark_rate_products'] = 100 * $marginInfos['margin_on_products'] / $marginInfos['pv_products'];
171 
172  if ($marginInfos['pa_services'] > 0)
173  $marginInfos['margin_rate_services'] = 100 * $marginInfos['margin_on_services'] / $marginInfos['pa_services'];
174  if ($marginInfos['pv_services'] > 0)
175  $marginInfos['mark_rate_services'] = 100 * $marginInfos['margin_on_services'] / $marginInfos['pv_services'];
176 
177  // if credit note, margin = -1 * (abs(selling_price) - buying_price)
178  //if ($marginInfos['pv_total'] < 0)
179  // $marginInfos['total_margin'] = -1 * (abs($marginInfos['pv_total']) - $marginInfos['pa_total']);
180  //else
181  $marginInfos['total_margin'] = $marginInfos['pv_total'] - $marginInfos['pa_total'];
182  if ($marginInfos['pa_total'] > 0)
183  $marginInfos['total_margin_rate'] = 100 * $marginInfos['total_margin'] / $marginInfos['pa_total'];
184  if ($marginInfos['pv_total'] > 0)
185  $marginInfos['total_mark_rate'] = 100 * $marginInfos['total_margin'] / $marginInfos['pv_total'];
186 
187  return $marginInfos;
188  }
