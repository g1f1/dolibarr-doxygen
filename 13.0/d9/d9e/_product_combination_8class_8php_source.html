479  public function updateProperties(Product $parent, User $user)
480  {
481  global $conf;
482 
483  $this->db->begin();
484 
485  $child = new Product($this->db);
486  $child->fetch($this->fk_product_child);
487 
488  $child->price_autogen = $parent->price_autogen;
489  $child->weight = $parent->weight;
490  if ($this->variation_weight) { // If we must add a delta on weight
491  $child->weight = ($child->weight ? $child->weight : 0) + $this->variation_weight;
492  }
493  $child->weight_units = $parent->weight_units;
494 
495  // Don't update the child label if the user has already modified it.
496  if ($child->label == $parent->label) {
497  // This will trigger only at variant creation time
498  $varlabel = $this->getCombinationLabel($this->fk_product_child);
499  $child->label = $parent->label.$varlabel; ;
500  }
501 
502 
503  if ($child->update($child->id, $user) > 0) {
504  $new_vat = $parent->tva_tx;
505  $new_npr = $parent->tva_npr;
506 
507  // MultiPrix
508  if (!empty($conf->global->PRODUIT_MULTIPRICES)) {
509  for ($i = 1; $i <= $conf->global->PRODUIT_MULTIPRICES_LIMIT; $i++)
510  {
511  if ($parent->multiprices[$i] != '' || isset($this->combination_price_levels[$i]->variation_price)) {
512  $new_type = $parent->multiprices_base_type[$i];
513  $new_min_price = $parent->multiprices_min[$i];
514  $variation_price = doubleval(!isset($this->combination_price_levels[$i]->variation_price) ? $this->variation_price : $this->combination_price_levels[$i]->variation_price);
515  $variation_price_percentage = doubleval(!isset($this->combination_price_levels[$i]->variation_price_percentage) ? $this->variation_price_percentage : $this->combination_price_levels[$i]->variation_price_percentage);
516 
517  if ($parent->prices_by_qty_list[$i]) {
518  $new_psq = 1;
519  } else {
520  $new_psq = 0;
521  }
522 
523  if ($new_type == 'TTC') {
524  $new_price = $parent->multiprices_ttc[$i];
525  } else {
526  $new_price = $parent->multiprices[$i];
527  }
528 
529  if ($variation_price_percentage) {
530  if ($new_price != 0) {
531  $new_price *= 1 + ($variation_price / 100);
532  }
533  } else {
534  $new_price += $variation_price;
535  }
536 
537  $child->updatePrice($new_price, $new_type, $user, $new_vat, $new_min_price, $i, $new_npr, $new_psq, 0, array(), $parent->default_vat_code);
538  }
539  }
540  } else {
541  $new_type = $parent->price_base_type;
542  $new_min_price = $parent->price_min;
543  $new_psq = $parent->price_by_qty;
544 
545  if ($new_type == 'TTC') {
546  $new_price = $parent->price_ttc;
547  } else {
548  $new_price = $parent->price;
549  }
550 
551  if ($this->variation_price_percentage) {
552  if ($new_price != 0) {
553  $new_price *= 1 + ($this->variation_price / 100);
554  }
555  } else {
556  $new_price += $this->variation_price;
557  }
558 
559  $child->updatePrice($new_price, $new_type, $user, $new_vat, $new_min_price, 1, $new_npr, $new_psq);
560  }
561 
562  $this->db->commit();
563 
564  return 1;
565  }
566 
567  $this->db->rollback();
568  return -1;
569  }

677  public function createProductCombination(User $user, Product $product, array $combinations, array $variations, $price_var_percent = false, $forced_pricevar = false, $forced_weightvar = false, $forced_refvar = false, $ref_ext = '')
678  {
679  global $conf;
680 
681  require_once DOL_DOCUMENT_ROOT.'/variants/class/ProductAttribute.class.php';
682  require_once DOL_DOCUMENT_ROOT.'/variants/class/ProductAttributeValue.class.php';
683 
684  $this->db->begin();
685 
686  $price_impact = array(1=>0); // init level price impact
687 
688  $forced_refvar = trim($forced_refvar);
689 
690  if (!empty($forced_refvar) && $forced_refvar != $product->ref) {
691  $existingProduct = new Product($this->db);
692  $result = $existingProduct->fetch('', $forced_refvar);
693  if ($result > 0) {
694  $newproduct = $existingProduct;
695  } else {
696  $existingProduct = false;
697  $newproduct = clone $product;
698  $newproduct->ref = $forced_refvar;
699  }
700  } else {
701  $forced_refvar = false;
702  $existingProduct = false;
703  $newproduct = clone $product;
704  }
705 
706  //Final weight impact
707  $weight_impact = (float) $forced_weightvar; // If false, return 0
708 
709  //Final price impact
710  if (!is_array($forced_pricevar)) {
711  $price_impact[1] = (float) $forced_pricevar; // If false, return 0
712  }
713  else {
714  $price_impact = $forced_pricevar;
715  }
716 
717  $newcomb = new ProductCombination($this->db);
718  $existingCombination = $newcomb->fetchByProductCombination2ValuePairs($product->id, $combinations);
719 
720  if ($existingCombination) {
721  $newcomb = $existingCombination;
722  } else {
723  $newcomb->fk_product_parent = $product->id;
724 
725  // Create 1 entry into product_attribute_combination (1 entry for each combinations). This init also $newcomb->id
726  $result = $newcomb->create($user);
727  if ($result < 0) {
728  $this->error = $newcomb->error;
729  $this->errors = $newcomb->errors;
730  $this->db->rollback();
731  return -1;
732  }
733  }
734 
735  $prodattr = new ProductAttribute($this->db);
736  $prodattrval = new ProductAttributeValue($this->db);
737 
738  // $combination contains list of attributes pairs key->value. Example: array('id Color'=>id Blue, 'id Size'=>id Small, 'id Option'=>id val a, ...)
739  //var_dump($combinations);
740  foreach ($combinations as $currcombattr => $currcombval) {
741  //This was checked earlier, so no need to double check
742  $prodattr->fetch($currcombattr);
743  $prodattrval->fetch($currcombval);
744 
745  //If there is an existing combination, there is no need to duplicate the valuepair
746  if (!$existingCombination) {
747  $tmp = new ProductCombination2ValuePair($this->db);
748  $tmp->fk_prod_attr = $currcombattr;
749  $tmp->fk_prod_attr_val = $currcombval;
750  $tmp->fk_prod_combination = $newcomb->id;
751 
752  if ($tmp->create($user) < 0) { // Create 1 entry into product_attribute_combination2val
753  $this->error = $tmp->error;
754  $this->errors = $tmp->errors;
755  $this->db->rollback();
756  return -1;
757  }
758  }
759 
760  if ($forced_weightvar === false) {
761  $weight_impact += (float) price2num($variations[$currcombattr][$currcombval]['weight']);
762  }
763  if ($forced_pricevar === false) {
764  $price_impact[1] += (float) price2num($variations[$currcombattr][$currcombval]['price']);
765 
766  // Manage Price levels
767  if ($conf->global->PRODUIT_MULTIPRICES) {
768  for ($i = 2; $i <= $conf->global->PRODUIT_MULTIPRICES_LIMIT; $i++)
769  {
770  $price_impact[$i] += (float) price2num($variations[$currcombattr][$currcombval]['price']);
771  }
772  }
773  }
774 
775  if ($forced_refvar === false) {
776  if (isset($conf->global->PRODUIT_ATTRIBUTES_SEPARATOR)) {
777  $newproduct->ref .= $conf->global->PRODUIT_ATTRIBUTES_SEPARATOR.$prodattrval->ref;
778  } else {
779  $newproduct->ref .= '_'.$prodattrval->ref;
780  }
781  }
782 
783  //The first one should not contain a linebreak
784  if ($newproduct->description) {
785  $newproduct->description .= '<br>';
786  }
787  $newproduct->description .= '<strong>'.$prodattr->label.':</strong> '.$prodattrval->value;
788  }
789 
790  $newcomb->variation_price_percentage = $price_var_percent;
791  $newcomb->variation_price = $price_impact[1];
792  $newcomb->variation_weight = $weight_impact;
793  $newcomb->variation_ref_ext = $this->db->escape($ref_ext);
